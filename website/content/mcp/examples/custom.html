自定义MCP服务器

来源说明：本文展示如何开发自定义 MCP 服务器。

可靠性评级：A 官方 SDK 文档

开发概述

MCP 提供了多种语言的 SDK，帮助开发者快速创建自定义服务器。

支持的 SDK

  • TypeScript/JavaScript (@modelcontextprotocol/sdk)
  • Python (mcp)
  • Go (go-mcp)
  • Rust (mcp-rust-sdk)

TypeScript 服务器开发

项目初始化

# 创建项目
mkdir my-mcp-server && cd my-mcp-server

# 初始化 npm
npm init -y

# 安装 SDK
npm install @modelcontextprotocol/sdk

服务器代码

// src/index.ts
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';

// 创建服务器实例
const server = new Server(
  {
    name: 'my-custom-server',
    version: '1.0.0',
  },
  {
    capabilities: {
      tools: {},
      resources: {},
    },
  }
);

// 注册工具列表
server.setRequestHandler(ListToolsRequestSchema, async () => {
  return {
    tools: [
      {
        name: 'hello',
        description: '返回问候语',
        inputSchema: {
          type: 'object',
          properties: {
            name: {
              type: 'string',
              description: '用户名',
            },
          },
          required: ['name'],
        },
      },
    ],
  };
});

// 处理工具调用
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  if (request.params.name === 'hello') {
    const { name } = request.params.arguments;
    return {
      content: [
        {
          type: 'text',
          text: `你好，${name}！欢迎使用自定义 MCP 服务器。`,
        },
      ],
    };
  }
  throw new Error('Unknown tool');
});

// 启动服务器
const transport = new StdioServerTransport();
await server.connect(transport);

编译和运行

# 编译 TypeScript
npx tsc

# 运行服务器
node dist/index.js

Python 服务器开发

安装 SDK

# 安装 MCP Python SDK
pip install mcp

服务器代码

# server.py
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent

# 创建服务器
server = Server("my-python-server")

# 注册工具
@server.list_tools()
async def list_tools():
    return [
        Tool(
            name="calculate",
            description="执行数学计算",
            inputSchema={
                "type": "object",
                "properties": {
                    "expression": {
                        "type": "string",
                        "description": "数学表达式"
                    }
                },
                "required": ["expression"]
            }
        )
    ]

# 处理工具调用
@server.call_tool()
async def call_tool(name: str, arguments: dict):
    if name == "calculate":
        expression = arguments["expression"]
        try:
            result = eval(expression)
            return [TextContent(type="text", text=f"结果: {result}")]
        except Exception as e:
            return [TextContent(type="text", text=f"错误: {str(e)}")]
    raise ValueError(f"Unknown tool: {name}")

# 运行服务器
async def main():
    async with stdio_server() as (read_stream, write_stream):
        await server.run(read_stream, write_stream)

if __name__ == "__main__":
    import asyncio
    asyncio.run(main())

添加资源支持

定义资源

// 注册资源列表
server.setRequestHandler(ListResourcesRequestSchema, async () => {
  return {
    resources: [
      {
        uri: 'custom://config',
        name: '配置信息',
        description: '服务器配置',
        mimeType: 'application/json',
      },
    ],
  };
});

// 处理资源读取
server.setRequestHandler(ReadResourceRequestSchema, async (request) => {
  if (request.params.uri === 'custom://config') {
    return {
      contents: [
        {
          uri: 'custom://config',
          mimeType: 'application/json',
          text: JSON.stringify({
            version: '1.0.0',
            features: ['tools', 'resources'],
          }),
        },
      ],
    };
  }
  throw new Error('Resource not found');
});

配置使用

添加到 Claude Code

// ~/.claude/mcp-config.json
{
  "servers": {
    "my-custom": {
      "command": "node",
      "args": ["/path/to/my-server/dist/index.js"],
      "env": {
        "LOG_LEVEL": "debug"
      }
    }
  }
}

测试服务器

# 列出工具
> /mcp tools my-custom

# 调用工具
> /mcp call my-custom.hello --name "张三"

错误处理

try {
  // 执行操作
  const result = await someOperation();
  return { content: [{ type: 'text', text: result }] };
} catch (error) {
  return {
    content: [{
      type: 'text',
      text: `错误: ${error.message}`
    }],
    isError: true
  };
}
