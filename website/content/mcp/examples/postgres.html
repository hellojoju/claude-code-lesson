PostgreSQL集成案例

来源说明：本文展示一个完整的 PostgreSQL MCP 集成案例。

可靠性评级：B 社区实践

环境准备

安装 PostgreSQL MCP 服务器

# 使用 npm 安装
npm install -g @modelcontextprotocol/server-postgres

# 验证安装
mcp-server-postgres --version

准备数据库

-- 创建示例数据库
CREATE DATABASE shop_db;

-- 创建用户表
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  email VARCHAR(255) UNIQUE NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建订单表
CREATE TABLE orders (
  id SERIAL PRIMARY KEY,
  user_id INTEGER REFERENCES users(id),
  total DECIMAL(10, 2) NOT NULL,
  status VARCHAR(50) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 插入测试数据
INSERT INTO users (name, email) VALUES
  ('张三', 'zhangsan@example.com'),
  ('李四', 'lisi@example.com'),
  ('王五', 'wangwu@example.com');

INSERT INTO orders (user_id, total, status) VALUES
  (1, 99.99, 'completed'),
  (1, 149.99, 'pending'),
  (2, 79.99, 'completed'),
  (3, 199.99, 'pending');

配置 MCP

MCP 配置文件

// ~/.claude/mcp-config.json
{
  "servers": {
    "shop-db": {
      "command": "mcp-server-postgres",
      "env": {
        "DATABASE_URL": "postgresql://postgres:password@localhost:5432/shop_db"
      },
      "permissions": {
        "allow": ["SELECT", "INSERT", "UPDATE"],
        "deny": ["DELETE", "DROP", "TRUNCATE"],
        "maxRows": 1000
      }
    }
  }
}

使用示例

查询数据

# 简单查询
> 查询所有用户

# Claude 生成的 SQL
SELECT * FROM users;

# 复杂查询
> 查询每个用户的订单数量和总金额

# Claude 生成的 SQL
SELECT 
  u.name,
  u.email,
  COUNT(o.id) as order_count,
  SUM(o.total) as total_amount
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.name, u.email;

数据分析

# 分析订单状态
> 分析订单状态分布

# Claude 生成的 SQL
SELECT 
  status,
  COUNT(*) as count,
  SUM(total) as total_amount
FROM orders
GROUP BY status
ORDER BY count DESC;

高级用法

事务处理

# 创建订单事务
> 为用户 ID 1 创建一个新订单，金额 299.99

# Claude 会使用事务
BEGIN;
INSERT INTO orders (user_id, total, status)
VALUES (1, 299.99, 'pending');
COMMIT;

复杂查询

# 统计分析
> 统计最近一个月每天的订单数量和金额

# Claude 生成的 SQL
SELECT 
  DATE(created_at) as date,
  COUNT(*) as order_count,
  SUM(total) as daily_total
FROM orders
WHERE created_at >= NOW() - INTERVAL '1 month'
GROUP BY DATE(created_at)
ORDER BY date DESC;
