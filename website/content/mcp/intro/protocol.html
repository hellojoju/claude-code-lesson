MCP协议规范

来源说明：本文内容基于 MCP 官方规范文档整理。

可靠性评级：A 官方规范

协议概述

MCP 协议定义了客户端和服务器之间的通信规范，包括消息格式、传输方式、能力协商等。

协议版本

当前版本: 2024-11-05
协议标识: mcp/1.0

传输层

支持的传输方式

传输方式 说明 适用场景
stdio 标准输入输出 本地进程通信
HTTP HTTP 请求响应 远程服务
WebSocket 双向实时通信 实时交互

stdio 传输

{
  "transport": {
    "type": "stdio",
    "command": "mcp-server-postgres",
    "args": ["--port", "5432"]
  }
}

HTTP 传输

{
  "transport": {
    "type": "http",
    "url": "http://localhost:3000/mcp",
    "headers": {
      "Authorization": "Bearer token"
    }
  }
}

消息格式

请求消息

{
  "jsonrpc": "2.0",
  "id": "request-001",
  "method": "tools/call",
  "params": {
    "name": "query",
    "arguments": {
      "sql": "SELECT * FROM users"
    }
  }
}

响应消息

{
  "jsonrpc": "2.0",
  "id": "request-001",
  "result": {
    "content": [
      {
        "type": "text",
        "text": "[{\"id\": 1, \"name\": \"Alice\"}]"
      }
    ]
  }
}

错误消息

{
  "jsonrpc": "2.0",
  "id": "request-001",
  "error": {
    "code": -32600,
    "message": "Invalid Request",
    "data": {
      "details": "Missing required parameter: sql"
    }
  }
}

资源规范

资源列表

// 请求
{
  "method": "resources/list"
}

// 响应
{
  "result": {
    "resources": [
      {
        "uri": "postgres://users",
        "name": "Users Table",
        "description": "用户表",
        "mimeType": "application/json"
      }
    ]
  }
}

资源读取

// 请求
{
  "method": "resources/read",
  "params": {
    "uri": "postgres://users"
  }
}

// 响应
{
  "result": {
    "contents": [
      {
        "uri": "postgres://users",
        "mimeType": "application/json",
        "text": "[{\"id\": 1, \"name\": \"Alice\"}]"
      }
    ]
  }
}

工具规范

工具列表

// 请求
{
  "method": "tools/list"
}

// 响应
{
  "result": {
    "tools": [
      {
        "name": "query",
        "description": "执行 SQL 查询",
        "inputSchema": {
          "type": "object",
          "properties": {
            "sql": {
              "type": "string",
              "description": "SQL 查询语句"
            }
          },
          "required": ["sql"]
        }
      }
    ]
  }
}

工具调用

// 请求
{
  "method": "tools/call",
  "params": {
    "name": "query",
    "arguments": {
      "sql": "SELECT * FROM users LIMIT 10"
    }
  }
}

提示词规范

提示词列表

{
  "result": {
    "prompts": [
      {
        "name": "analyze-table",
        "description": "分析数据库表结构",
        "arguments": [
          {
            "name": "table",
            "description": "表名",
            "required": true
          }
        ]
      }
    ]
  }
}

获取提示词

// 请求
{
  "method": "prompts/get",
  "params": {
    "name": "analyze-table",
    "arguments": {
      "table": "users"
    }
  }
}

// 响应
{
  "result": {
    "description": "分析 users 表",
    "messages": [
      {
        "role": "user",
        "content": {
          "type": "text",
          "text": "请分析 users 表的结构和数据..."
        }
      }
    ]
  }
}
