服务器实现

// src/server.ts
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';

const server = new Server(
    {
        name: 'my-custom-server',
        version: '1.0.0'
    },
    {
        capabilities: {
            tools: {},
            resources: {}
        }
    }
);

// 注册工具列表处理器
server.setRequestHandler(ListToolsRequestSchema, async () => {
    return {
        tools: [
            {
                name: 'my_custom_tool',
                description: '执行自定义操作',
                inputSchema: {
                    type: 'object',
                    properties: {
                        input: {
                            type: 'string',
                            description: '输入参数'
                        }
                    },
                    required: ['input']
                }
            }
        ]
    };
});

// 注册工具调用处理器
server.setRequestHandler(CallToolRequestSchema, async (request) => {
    const { name, arguments: args } = request.params;

    switch (name) {
        case 'my_custom_tool':
            return {
                content: [{
                    type: 'text',
                    text: await processCustomTool(args.input)
                }]
            };

        default:
            throw new Error(`Unknown tool: ${name}`);
    }
});

// 注册资源列表处理器
server.setRequestHandler(ListResourcesRequestSchema, async () => {
    return {
        resources: [
            {
                uri: 'custom://data',
                name: '自定义数据',
                mimeType: 'application/json'
            }
        ]
    };
});

// 注册资源读取处理器
server.setRequestHandler(ReadResourceRequestSchema, async (request) => {
    const { uri } = request.params;

    if (uri === 'custom://data') {
        return {
            contents: [{
                uri,
                mimeType: 'application/json',
                text: JSON.stringify({ data: 'custom data' })
            }]
        };
    }

    throw new Error(`Unknown resource: ${uri}`);
});

// 启动服务器
async function main() {
    const transport = new StdioServerTransport();
    await server.connect(transport);
}

main().catch(console.error);

工具实现最佳实践

// src/tools/my-tool.ts
export interface MyToolParams {
    input: string;
    options?: {
        verbose?: boolean;
        timeout?: number;
    };
}

export async function executeMyTool(params: MyToolParams): Promise<string> {
    // 1. 验证参数
    validateParams(params);

    // 2. 执行操作
    try {
        const result = await performOperation(params);

        // 3. 格式化输出
        return formatResult(result);
    } catch (error) {
        // 4. 错误处理
        return formatError(error);
    }
}

function validateParams(params: MyToolParams): void {
    if (!params.input || typeof params.input !== 'string') {
        throw new Error('input must be a non-empty string');
    }

    if (params.options?.timeout && params.options.timeout < 0) {
        throw new Error('timeout must be positive');
    }
}

function formatResult(result: any): string {
    // 返回结构化的结果
    return JSON.stringify({
        success: true,
        data: result,
        timestamp: Date.now()
    }, null, 2);
}

function formatError(error: Error): string {
    return JSON.stringify({
        success: false,
        error: {
            message: error.message,
            type: error.constructor.name
        }
    }, null, 2);
}

# API Designer Skill

You are an expert API designer specializing in RESTful architectures.

## Your Task

Design clean, intuitive, and well-documented APIs following industry best practices.

## Design Principles

1. **Resource-Oriented**: Design around resources, not actions
2. **Consistent Naming**: Use plural nouns for resources
3. **Proper HTTP Methods**: Use methods semantically
4. **Versioning**: Include version in URL path
5. **Error Handling**: Return meaningful error messages

## Output Format

Generate API specifications in the requested format:

### For OpenAPI:
```yaml
openapi: 3.0.0
info:
  title: API Title
  version: 1.0.0
paths:
  /resources:
    get:
      summary: List resources
      ...
```

## Process

1. Understand the requirements
2. Identify resources and relationships
3. Design endpoints and methods
4. Define request/response schemas
5. Add error responses
6. Generate documentation

Hook 脚本示例

// scripts/security-check.js
const fs = require('fs');

async function main() {
    // 从 stdin 读取上下文
    const input = JSON.parse(await readStdin());
    
    const { tool, params } = input.data;

    if (tool !== 'write_file') {
        process.exit(0);
    }

    const content = params.content;
    const path = params.path;

    // 安全检查
    const issues = [];

    // 检查敏感信息
    if (content.includes('password') || content.includes('secret')) {
        issues.push('Possible sensitive data in code');
    }

    // 检查 SQL 注入风险
    if (content.includes('${') && content.includes('SQL')) {
        issues.push('Possible SQL injection risk');
    }

    // 检查硬编码的凭证
    const credentialPatterns = [
        /api[_-]?key\s*=\s*['"][^'"]+['"]/i,
        /token\s*=\s*['"][^'"]+['"]/i,
        /password\s*=\s*['"][^'"]+['"]/i
    ];

    for (const pattern of credentialPatterns) {
        if (pattern.test(content)) {
            issues.push('Hardcoded credentials detected');
            break;
        }
    }

    if (issues.length > 0) {
        // 输出警告但不阻止
        console.log(JSON.stringify({
            modify: {
                ...input.data,
                warnings: issues
            }
        }));
    }
}

function readStdin() {
    return new Promise((resolve) => {
        let data = '';
        process.stdin.on('data', chunk => data += chunk);
        process.stdin.on('end', () => resolve(data));
    });
}

main().catch(console.error);

最佳实践总结

开发原则

  • 单一职责：每个扩展只做一件事
  • 文档完善：提供清晰的使用说明
  • 错误处理：优雅地处理各种异常
  • 性能优先：避免阻塞主流程

安全考虑

  • 验证所有输入参数
  • 使用环境变量存储敏感信息
  • 限制文件系统访问范围
  • 实现适当的权限检查

维护建议

  • 使用语义化版本控制
  • 编写完整的测试用例
  • 保持向后兼容性
  • 及时更新依赖
