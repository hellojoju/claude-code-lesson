5.1 消息历史结构

// 推测的消息历史结构
interface ConversationHistory {
    messages: Message[];
    metadata: {
        sessionId: string;
        createdAt: number;
        updatedAt: number;
        totalTokens: number;
    };
}

interface Message {
    role: 'user' | 'assistant';
    content: Content[];
    timestamp: number;
    metadata?: {
        cached?: boolean;
        compressed?: boolean;
    };
}

type Content = 
    | { type: 'text'; text: string }
    | { type: 'tool_use'; id: string; name: string; input: any }
    | { type: 'tool_result'; tool_use_id: string; content: string }

interface ContextWindow {
    maxTokens: number;

第六步：创建分析脚本

6.1 日志解析脚本

#!/usr/bin/env python3
# analyze_logs.py

import json
import sys
from collections import defaultdict
from datetime import datetime

def parse_logs(log_file):
    """解析 Claude Code 日志文件"""
    events = []
    
    with open(log_file, 'r') as f:
        for line in f:
            try:
                event = json.loads(line.strip())
                events.append(event)
            except json.JSONDecodeError:
                continue
    
    return events

def analyze_tool_usage(events):
    """分析工具使用情况"""
    tool_counts = defaultdict(int)
    tool_times = defaultdict(list)
    
    for event in events:
        if event.get('type') == 'tool_call':
            tool_name = event.get('tool')
            tool_counts[tool_name] += 1
            tool_times[tool_name].append(event.get('duration', 0))
    
    print("\n工具使用统计:")
    for tool, count in sorted(tool_counts.items(), key=lambda x: -x[1]):
        avg_time = sum(tool_times[tool]) / len(tool_times[tool])
        print(f"  {tool}: {count} 次, 平均耗时 {avg_time:.2f}ms")

def analyze_token_usage(events):
    """分析 Token 使用情况"""
    total_input = 0
    total_output = 0
    
    for event in events:
        if 'usage' in event:
            total_input += event['usage'].get('input_tokens', 0)
            total_output += event['usage'].get('output_tokens', 0)
    
    print(f"\nToken 使用统计:")
    print(f"  输入: {total_input:,}")
    print(f"  输出: {total_output:,}")
    print(f"  总计: {total_input + total_output:,}")

if __name__ == '__main__':
    if len(sys.argv) < 2:
        print("用法: python analyze_logs.py <log_file>")
        sys.exit(1)
    
    events = parse_logs(sys.argv[1])
    analyze_tool_usage(events)
    analyze_token_usage(events)

6.2 API 请求监控脚本

// monitor_api.js
const https = require('https');
const http = require('http');

// 代理服务器
const proxy = http.createServer((req, res) => {
    console.log(`[${new Date().toISOString()}] ${req.method} ${req.url}`);
    
    // 记录请求头
    console.log('Request Headers:', JSON.stringify(req.headers, null, 2));
    
    // 转发请求
    const options = {
        hostname: 'api.anthropic.com',
        port: 443,
        path: req.url,
        method: req.method,
        headers: req.headers
    };
    
    const proxyReq = https.request(options, (proxyRes) => {
        console.log('Response Status:', proxyRes.statusCode);
        res.writeHead(proxyRes.statusCode, proxyRes.headers);
        proxyRes.pipe(res);
    });
    
    req.pipe(proxyReq);
});

proxy.listen(8080, () => {
    console.log('代理服务器运行在 http://localhost:8080');
    console.log('设置环境变量: export HTTPS_PROXY=http://localhost:8080');
});

注意事项

安全警告

  • 不要在调试日志中暴露 API 密钥
  • 敏感文件内容可能被记录到日志
  • 生产环境不建议开启详细日志
  • 定期清理日志文件释放磁盘空间
