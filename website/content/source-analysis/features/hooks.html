Hooks 系统概述

Hooks 是 Claude Code 的事件监听机制，允许在特定事件发生时执行自定义代码。本节深入分析 Hooks 的设计原理和实现机制。

Hook 注册系统

Hook 定义

// 推测的 Hook 类型定义
interface Hook {
    id: string;
    event: HookEvent;
    handler: HookHandler;
    priority: number;
    enabled: boolean;
    conditions?: HookCondition[];
}

type HookEvent = 
    | 'PrePrompt'
    | 'PreToolUse'
    | 'PostToolUse'
    | 'PreResponse'
    | 'PostResponse'
    | 'OnError'
    | 'OnFileChange';

interface HookContext {
    event: HookEvent;
    timestamp: number;
    sessionId: string;
    data: any;
    metadata: Record<string, any>;
}

type HookHandler = (
    context: HookContext
) => Promise<HookResult | void>;

interface HookResult {
    // 修改数据
    modify?: any;
    // 阻止继续执行
    block?: boolean;
    // 阻止原因
    reason?: string;
    // 添加额外操作
    actions?: HookAction[];
}

interface HookCondition {
    type: 'tool' | 'file' | 'pattern' | 'expression';
    value: string | RegExp | ((ctx: HookContext) => boolean);
}

注册中心实现

// 推测的 Hook 注册中心
class HookRegistry {
    private hooks: Map<HookEvent, Hook[]> = new Map();
    private config: HooksConfig;

    constructor(config: HooksConfig) {
        this.config = config;
        this.loadFromConfig();
    }

    register(hook: Hook): void {
        if (!this.hooks.has(hook.event)) {
            this.hooks.set(hook.event, []);
        }

        const hooks = this.hooks.get(hook.event)!;
        hooks.push(hook);

        // 按优先级排序
        hooks.sort((a, b) => b.priority - a.priority);
    }

    unregister(hookId: string): void {
        for (const [event, hooks] of this.hooks) {
            const index = hooks.findIndex(h => h.id === hookId);
            if (index !== -1) {
                hooks.splice(index, 1);
            }
        }
    }

    getHooks(event: HookEvent): Hook[] {
        return this.hooks.get(event) || [];
    }

    private loadFromConfig(): void {
        for (const hookConfig of this.config.hooks) {
            this.register({
                id: hookConfig.id || generateId(),
                event: hookConfig.event,
                handler: this.createHandler(hookConfig),
                priority: hookConfig.priority || 0,
                enabled: hookConfig.enabled !== false,
                conditions: hookConfig.conditions
            });
        }
    }

    private createHandler(config: HookConfig): HookHandler {
        return async (context: HookContext) => {
            // Shell 命令类型
            if (config.type === 'command') {
                return this.executeCommand(config.command, context);
            }

            // JavaScript 函数类型
            if (config.type === 'function') {
                const fn = new Function('context', config.function);
                return fn(context);
            }

            // HTTP 请求类型
            if (config.type === 'http') {
                return this.executeHttp(config.http, context);
            }
        };
    }
}

Hook 执行器

执行流程

// 推测的 Hook 执行器实现
class HookExecutor {
    constructor(
        private registry: HookRegistry,
        private logger: Logger
    ) {}

    async execute(
        event: HookEvent, 
        data: any
    ): Promise<HookExecutionResult> {
        const hooks = this.registry.getHooks(event);
        const context: HookContext = {
            event,
            timestamp: Date.now(),
            sessionId: getCurrentSessionId(),
            data,
            metadata: {}
        };

        const result: HookExecutionResult = {
            modified: data,
            blocked: false,
            actions: [],
            errors: []
        };

        for (const hook of hooks) {
            if (!hook.enabled) continue;

            // 检查条件
            if (!this.matchesConditions(hook, context)) {
                continue;
            }

            try {
                const hookResult = await this.executeWithTimeout(
                    hook.handler,
                    context,
                    hook.timeout || 30000
                );

                if (hookResult) {
                    // 处理修改
                    if (hookResult.modify) {
                        result.modified = hookResult.modify;
                        context.data = hookResult.modify;
                    }

                    // 处理阻止
                    if (hookResult.block) {
                        result.blocked = true;
                        result.blockReason = hookResult.reason;
                        break;
                    }

                    // 收集额外操作
                    if (hookResult.actions) {
                        result.actions.push(...hookResult.actions);
                    }
                }

            } catch (error) {
                this.logger.error(`Hook ${hook.id} failed:`, error);
                result.errors.push({
                    hookId: hook.id,
                    error: error.message
                });

                // 根据配置决定是否继续
                if (hook.failStrategy === 'stop') {
                    break;
                }
            }
        }

        return result;
    }

    private matchesConditions(hook: Hook, context: HookContext): boolean {
        if (!hook.conditions || hook.conditions.length === 0) {
            return true;
        }

        return hook.conditions.every(condition => {
            switch (condition.type) {
                case 'tool':
                    return context.data.tool === condition.value;
                case 'file':
                    return this.matchFilePattern(
                        context.data.path, 
                        condition.value
                    );
                case 'pattern':
                    return this.matchPattern(
                        JSON.stringify(context.data),
                        condition.value
                    );
                case 'expression':
                    return (condition.value as Function)(context);
                default:
                    return true;
            }
        });
    }

    private async executeWithTimeout(
        handler: HookHandler,
        context: HookContext,
        timeout: number
    ): Promise<HookResult | void> {
        return Promise.race([
            handler(context),
            new Promise<never>((_, reject) => 
                setTimeout(() => reject(new TimeoutError()), timeout)
            )
        ]);
    }
}

PreToolUse Hook 示例

// 推测的 PreToolUse Hook 实现
const preToolUseHooks: Hook[] = [
    {
        id: 'validate-file-path',
        event: 'PreToolUse',
        priority: 100,
        handler: async (context) => {
            const { tool, params } = context.data;

            // 验证文件路径
            if (tool === 'read_file' || tool === 'write_file') {
                const path = params.path;
                
                // 检查是否在允许的目录内
                if (!isPathAllowed(path)) {
                    return {
                        block: true,
                        reason: `Path not allowed: ${path}`
                    };
                }

                // 检查敏感文件
                if (isSensitiveFile(path)) {
                    return {
                        block: true,
                        reason: `Sensitive file access denied: ${path}`
                    };
                }
            }
        }
    },

    {
        id: 'log-tool-call',
        event: 'PreToolUse',
        priority: 0,
        handler: async (context) => {
            const { tool, params } = context.data;
            logToolCall(tool, params);
        }
    }
];

错误处理

// 推测的错误处理 Hook
const errorHooks: Hook[] = [
    {
        id: 'global-error-handler',
        event: 'OnError',
        priority: 100,
        handler: async (context) => {
            const { error, tool, params } = context.data;

            // 记录错误
            await logError({
                error: error.message,
                stack: error.stack,
                tool,
                params,
                timestamp: Date.now()
            });

            // 发送通知
            if (error.severity === 'high') {
                await sendAlert({
                    type: 'error',
                    message: error.message,
                    context: { tool, params }
                });
            }

            // 尝试恢复
            if (isRecoverable(error)) {
                return {
                    actions: [{
                        type: 'retry',
                        params: { maxAttempts: 3 }
                    }]
                };
            }
        }
    }

性能考虑

性能影响

  • 每个 Hook 都会增加执行延迟
  • 避免在 Hook 中执行耗时操作
  • 使用条件过滤减少不必要的执行
  • 设置合理的超时时间
// 性能优化示例
const optimizedHook: Hook = {
    id: 'fast-validation',
    event: 'PreToolUse',
    
    // 快速条件检查
    conditions: [
        { type: 'tool', value: 'write_file' },
        { type: 'file', value: '*.ts' }
    ],
    
    handler: async (context) => {
        // 使用缓存
        const cached = validationCache.get(context.data.params.path);
        if (cached) return cached;
        
        // 快速验证
        const result = await quickValidate(context.data.params.content);
        validationCache.set(context.data.params.path, result);
        
        return result;
    },
    
    // 设置超时
    timeout: 5000
};
