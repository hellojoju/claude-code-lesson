客户端实现

连接管理

// 推测的 MCP 客户端实现
class MCPClient {
    private transport: Transport;
    private pendingRequests: Map<string, PendingRequest> = new Map();
    private tools: Map<string, ToolDefinition> = new Map();
    private resources: Map<string, ResourceDefinition> = new Map();
    private capabilities: ServerCapabilities | null = null;

    async connect(config: ServerConfig): Promise<void> {
        // 创建传输层
        this.transport = this.createTransport(config);

        // 设置消息处理器
        this.transport.onMessage(this.handleMessage.bind(this));

        // 建立连接
        await this.transport.connect();

        // 初始化协议
        await this.initialize();
    }

    private createTransport(config: ServerConfig): Transport {
        switch (config.type) {
            case 'stdio':
                return new StdioTransport(config.command, config.args);
            case 'sse':
                return new SSETransport(config.url);
            case 'websocket':
                return new WebSocketTransport(config.url);
            default:
                throw new Error(`Unknown transport: ${config.type}`);
        }
    }

    private async initialize(): Promise<void> {
        const response = await this.sendRequest<InitializeResponse>(
            'initialize',
            {
                protocolVersion: '2024-11-05',
                capabilities: {
                    tools: {},
                    resources: {}
                },
                clientInfo: {
                    name: 'claude-code',
                    version: '1.0.0'
                }
            }
        );

        this.capabilities = response.capabilities;

        // 发送 initialized 通知
        this.sendNotification('notifications/initialized', {});

        // 加载工具和资源
        await this.loadTools();
        await this.loadResources();
    }

    private handleMessage(message: JSONRPCMessage): void {
        if ('id' in message) {
            // 响应消息
            const pending = this.pendingRequests.get(message.id);
            if (pending) {
                if ('result' in message) {
                    pending.resolve(message.result);
                } else {
                    pending.reject(new MCPError(message.error));
                }
                this.pendingRequests.delete(message.id);
            }
        } else {
            // 通知消息
            this.handleNotification(message);
        }
    }

    async callTool(
        name: string, 
        args: Record<string, any>
    ): Promise<CallToolResponse> {
        if (!this.tools.has(name)) {
            throw new Error(`Unknown tool: ${name}`);
        }

        return this.sendRequest<CallToolResponse>('tools/call', {
            name,
            arguments: args
        });
    }

    private sendRequest<T>(
        method: string, 
        params: any
    ): Promise<T> {
        return new Promise((resolve, reject) => {
            const id = this.generateId();
            
            this.pendingRequests.set(id, { resolve, reject });

            this.transport.send({
                jsonrpc: '2.0',
                id,
                method,
                params
            });

            // 超时处理
            setTimeout(() => {
                if (this.pendingRequests.has(id)) {
                    this.pendingRequests.delete(id);
                    reject(new TimeoutError(`Request ${method} timed out`));
                }
            }, 30000);
        });
    }
}

传输层实现

// Stdio 传输层
class StdioTransport implements Transport {
    private process: ChildProcess;
    private buffer: string = '';
    private messageHandler: ((msg: JSONRPCMessage) => void) | null = null;

    constructor(
        private command: string, 
        private args: string[]
    ) {}

    async connect(): Promise<void> {
        this.process = spawn(this.command, this.args, {
            stdio: ['pipe', 'pipe', 'pipe']
        });

        this.process.stdout.on('data', (data) => {
            this.handleData(data.toString());
        });

        this.process.stderr.on('data', (data) => {
            console.error(`MCP Server stderr: ${data}`);
        });

        this.process.on('close', (code) => {
            console.log(`MCP Server exited with code ${code}`);
        });
    }

    private handleData(data: string): void {
        this.buffer += data;

        // 解析完整的 JSON 消息
        const lines = this.buffer.split('\n');
        this.buffer = lines.pop() || '';

        for (const line of lines) {
            if (line.trim()) {
                try {
                    const message = JSON.parse(line);
                    this.messageHandler?.(message);
                } catch (e) {
                    console.error('Failed to parse message:', line);
                }
            }
        }
    }

    send(message: JSONRPCMessage): void {
        const json = JSON.stringify(message);
        this.process.stdin.write(json + '\n');
    }

    onMessage(handler: (msg: JSONRPCMessage) => void): void {
        this.messageHandler = handler;
    }

    async close(): Promise<void> {
        this.process.kill();
    }
}

基础服务器

// MCP 服务器基类
class MCPServer {
    private tools: Map<string, ToolHandler> = new Map();
    private resources: Map<string, ResourceHandler> = new Map();
    private transport: Transport;

    constructor(private serverInfo: ServerInfo) {}

    registerTool(
        name: string, 
        definition: ToolDefinition, 
        handler: ToolHandler
    ): void {
        this.tools.set(name, { definition, handler });
    }

    async start(transport: Transport): Promise<void> {
        this.transport = transport;
        transport.onMessage(this.handleMessage.bind(this));
        await transport.connect();
    }

    private async handleMessage(message: JSONRPCMessage): Promise<void> {
        if ('method' in message) {
            const response = await this.handleRequest(message);
            if (response && 'id' in message) {
                this.transport.send({
                    jsonrpc: '2.0',
                    id: message.id,
                    ...response
                });
            }
        }
    }

    private async handleRequest(
        request: JSONRPCRequest
    ): Promise<{ result?: any; error?: any } | null> {
        try {
            switch (request.method) {
                case 'initialize':
                    return { result: this.handleInitialize(request.params) };

                case 'tools/list':
                    return { result: this.handleListTools() };

                case 'tools/call':
                    return { result: await this.handleCallTool(request.params) };

                case 'resources/list':
                    return { result: this.handleListResources() };

                case 'resources/read':
                    return { result: await this.handleReadResource(request.params) };

                default:
                    return {
                        error: {
                            code: -32601,
                            message: `Method not found: ${request.method}`
                        }
                    };
            }
        } catch (error) {
            return {
                error: {
                    code: -32603,
                    message: error.message
                }
            };
        }
    }

    private handleInitialize(params: InitializeRequest): InitializeResponse {
        return {
            protocolVersion: '2024-11-05',
            capabilities: {
                tools: { listChanged: true },
                resources: { subscribe: true, listChanged: true }
            },
            serverInfo: this.serverInfo
        };
    }

    private handleListTools(): ListToolsResponse {
        return {
            tools: Array.from(this.tools.entries()).map(([name, { definition }]) => ({
                name,
                ...definition
            }))
        };
    }

    private async handleCallTool(params: CallToolRequest): Promise<CallToolResponse> {
        const { handler } = this.tools.get(params.name) || {};
        
        if (!handler) {
            throw new Error(`Unknown tool: ${params.name}`);
        }

        const result = await handler(params.arguments);
        
        return {
            content: [{ type: 'text', text: result }]
        };
    }
}

工具集成

将 MCP 工具映射到 Claude Code

// 推测的工具集成实现
class MCPToolAdapter {
    private mcpClients: Map<string, MCPClient> = new Map();

    async loadMCPServers(configs: ServerConfig[]): Promise<void> {
        for (const config of configs) {
            const client = new MCPClient();
            await client.connect(config);
            this.mcpClients.set(config.name, client);
        }
    }

    getToolDefinitions(): ToolDefinition[] {
        const tools: ToolDefinition[] = [];

        for (const [serverName, client] of this.mcpClients) {
            for (const [toolName, tool] of client.getTools()) {
                tools.push({
                    name: `mcp_${serverName}_${toolName}`,
                    description: tool.description,
                    input_schema: tool.inputSchema,
                    // 添加服务器信息
                    annotations: {
                        server: serverName,
                        originalName: toolName
                    }
                });
            }
        }

        return tools;
    }

    async executeTool(
        name: string, 
        args: Record<string, any>
    ): Promise<any> {
        // 解析工具名称
        const [prefix, serverName, ...toolNameParts] = name.split('_');
        const toolName = toolNameParts.join('_');

        const client = this.mcpClients.get(serverName);
        if (!client) {
            throw new Error(`Unknown MCP server: ${serverName}`);
        }

        const response = await client.callTool(toolName, args);

        // 处理响应内容
        return this.processToolResponse(response);
    }

    private processToolResponse(response: CallToolResponse): any {
        const contents: string[] = [];

        for (const block of response.content) {
            if (block.type === 'text') {
                contents.push(block.text);
            } else if (block.type === 'image') {
                contents.push(`[Image: ${block.mimeType}]`);
            } else if (block.type === 'resource') {
                contents.push(`[Resource: ${block.resource.uri}]`);
            }
        }

        return contents.join('\n');
    }
}

资源系统

资源订阅

// 资源订阅实现
class ResourceManager {
    private subscriptions: Map<string, Set<string>> = new Map();

    async subscribe(uri: string, clientId: string): Promise<void> {
        if (!this.subscriptions.has(uri)) {
            this.subscriptions.set(uri, new Set());
        }
        this.subscriptions.get(uri)!.add(clientId);

        // 发送订阅请求
        await this.sendRequest('resources/subscribe', { uri });
    }

    async unsubscribe(uri: string, clientId: string): Promise<void> {
        const subscribers = this.subscriptions.get(uri);
        if (subscribers) {
            subscribers.delete(clientId);
            if (subscribers.size === 0) {
                await this.sendRequest('resources/unsubscribe', { uri });
                this.subscriptions.delete(uri);
            }
        }
    }

    handleResourceUpdate(notification: ResourceUpdatedNotification): void {
        const { uri, contents } = notification.params;

        // 通知所有订阅者
        const subscribers = this.subscriptions.get(uri);
        if (subscribers) {
            for (const clientId of subscribers) {
                this.notifyClient(clientId, {
                    type: 'resource/updated',
                    uri,
                    contents
                });
            }
        }
    }
}
