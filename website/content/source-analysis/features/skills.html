Skill 定义结构

Skill 元数据

// 推测的 Skill 类型定义
interface Skill {
    // 元数据
    id: string;
    name: string;
    version: string;
    description: string;
    author?: string;
    tags: string[];

    // 触发配置
    triggers: SkillTrigger[];

    // 能力定义
    capabilities: SkillCapability[];

    // 执行配置
    execution: {
        mode: 'single' | 'interactive' | 'autonomous';
        maxSteps?: number;
        timeout?: number;
    };

    // 依赖
    dependencies?: {
        skills?: string[];
        tools?: string[];
        mcpServers?: string[];
    };

    // 提示词模板
    prompts: {
        system?: string;
        user?: string;
        examples?: Example[];
    };

    // 配置模式
    configSchema?: JSONSchema;
}

interface SkillTrigger {
    type: 'command' | 'pattern' | 'intent' | 'file';
    value: string | RegExp;
    priority?: number;
}

interface SkillCapability {
    name: string;
    description: string;
    inputSchema?: JSONSchema;
    outputSchema?: JSONSchema;
}

Skill 加载器

发现和加载

// 推测的 Skill 加载器实现
class SkillLoader {
    private searchPaths: string[];
    private loadedSkills: Map<string, Skill> = new Map();

    constructor(searchPaths: string[]) {
        this.searchPaths = searchPaths;
    }

    async loadAll(): Promise<Skill[]> {
        const skills: Skill[] = [];

        for (const searchPath of this.searchPaths) {
            const discovered = await this.discoverSkills(searchPath);
            
            for (const skillPath of discovered) {
                try {
                    const skill = await this.loadSkill(skillPath);
                    skills.push(skill);
                    this.loadedSkills.set(skill.id, skill);
                } catch (error) {
                    console.error(`Failed to load skill from ${skillPath}:`, error);
                }
            }
        }

        // 验证依赖
        this.validateDependencies(skills);

        return skills;
    }

    private async discoverSkills(basePath: string): Promise<string[]> {
        const skillPaths: string[] = [];
        const entries = await fs.readdir(basePath, { withFileTypes: true });

        for (const entry of entries) {
            if (!entry.isDirectory()) continue;

            const skillDir = path.join(basePath, entry.name);
            const skillFile = path.join(skillDir, 'skill.yaml');

            if (await this.fileExists(skillFile)) {
                skillPaths.push(skillDir);
            }
        }

        return skillPaths;
    }

    private async loadSkill(skillPath: string): Promise<Skill> {
        const skillFile = path.join(skillPath, 'skill.yaml');
        const content = await fs.readFile(skillFile, 'utf-8');
        const config = yaml.parse(content);

        // 验证 Skill 定义
        this.validateSkillDefinition(config);

        // 加载附加资源
        const skill: Skill = {
            ...config,
            basePath: skillPath
        };

        // 加载提示词文件
        if (config.prompts?.systemFile) {
            skill.prompts.system = await fs.readFile(
                path.join(skillPath, config.prompts.systemFile),
                'utf-8'
            );
        }

        // 加载示例
        if (config.prompts?.examplesDir) {
            skill.prompts.examples = await this.loadExamples(
                path.join(skillPath, config.prompts.examplesDir)
            );
        }

        return skill;
    }

    private validateSkillDefinition(config: any): void {
        const required = ['id', 'name', 'version', 'description'];
        
        for (const field of required) {
            if (!config[field]) {
                throw new Error(`Missing required field: ${field}`);
            }
        }

        // 验证版本格式
        if (!/^\d+\.\d+\.\d+$/.test(config.version)) {
            throw new Error('Invalid version format. Expected semver.');
        }
    }
}

Skill 执行器

执行模型

// 推测的 Skill 执行器实现
class SkillRunner {
    constructor(
        private llmClient: LLMClient,
        private toolRegistry: ToolRegistry,
        private contextManager: ContextManager
    ) {}

    async run(
        skill: Skill, 
        input: SkillInput
    ): Promise<SkillOutput> {
        // 创建执行上下文
        const context = await this.createContext(skill, input);

        // 构建系统提示词
        const systemPrompt = this.buildSystemPrompt(skill, context);

        // 构建用户消息
        const userMessage = this.buildUserMessage(skill, input);

        // 执行 Skill
        const result = await this.executeWithMode(
            skill,
            systemPrompt,
            userMessage,
            context
        );

        return result;
    }

    private async executeWithMode(
        skill: Skill,
        systemPrompt: string,
        userMessage: string,
        context: ExecutionContext
    ): Promise<SkillOutput> {
        switch (skill.execution.mode) {
            case 'single':
                return this.executeSingle(skill, systemPrompt, userMessage, context);

            case 'interactive':
                return this.executeInteractive(skill, systemPrompt, userMessage, context);

            case 'autonomous':
                return this.executeAutonomous(skill, systemPrompt, userMessage, context);

            default:
                throw new Error(`Unknown execution mode: ${skill.execution.mode}`);
        }
    }

    private async executeSingle(
        skill: Skill,
        systemPrompt: string,
        userMessage: string,
        context: ExecutionContext
    ): Promise<SkillOutput> {
        // 单次执行，直接调用 LLM
        const response = await this.llmClient.chat({
            system: systemPrompt,
            messages: [{ role: 'user', content: userMessage }],
            maxTokens: 4096
        });

        return {
            success: true,
            content: response.content,
            metadata: {
                tokens: response.usage
            }
        };
    }

    private async executeInteractive(
        skill: Skill,
        systemPrompt: string,
        userMessage: string,
        context: ExecutionContext
    ): Promise<SkillOutput> {
        // 交互式执行，支持工具调用和用户交互
        const messages: Message[] = [
            { role: 'user', content: userMessage }
        ];

        let stepCount = 0;
        const maxSteps = skill.execution.maxSteps || 50;

        while (stepCount < maxSteps) {
            const response = await this.llmClient.chat({
                system: systemPrompt,
                messages,
                tools: this.getAvailableTools(skill),
                stream: true
            });

            // 处理流式响应
            for await (const chunk of response.stream) {
                if (chunk.type === 'text') {
                    // 输出文本
                    process.stdout.write(chunk.text);
                } else if (chunk.type === 'tool_use') {
                    // 执行工具
                    const result = await this.executeTool(chunk.tool, chunk.params);
                    messages.push({
                        role: 'user',
                        content: [{
                            type: 'tool_result',
                            tool_use_id: chunk.id,
                            content: result
                        }]
                    });
                }
            }

            // 检查是否完成
            if (response.stopReason === 'end_turn') {
                break;
            }

            stepCount++;
        }

        return {
            success: true,
            content: this.extractContent(messages),
            metadata: { steps: stepCount }
        };
    }

    private async executeAutonomous(
        skill: Skill,
        systemPrompt: string,
        userMessage: string,
        context: ExecutionContext
    ): Promise<SkillOutput> {
        // 自主执行，无需用户干预
        // 类似 Agent 的执行模式
        const agent = new SkillAgent(skill, this.llmClient, this.toolRegistry);
        return agent.run(userMessage, context);
    }
}

提示词模板系统

模板引擎

// 推测的模板引擎实现
class PromptTemplateEngine {
    private templates: Map<string, CompiledTemplate> = new Map();

    compile(template: string): CompiledTemplate {
        // 解析模板变量
        const variables = this.extractVariables(template);
        
        return {
            template,
            variables,
            render: (context: TemplateContext) => this.render(template, context)
        };
    }

    private extractVariables(template: string): string[] {
        const regex = /\{\{(\w+(?:\.\w+)*)\}\}/g;
        const variables: string[] = [];
        let match;

        while ((match = regex.exec(template)) !== null) {
            variables.push(match[1]);
        }

        return [...new Set(variables)];
    }

    private render(template: string, context: TemplateContext): string {
        return template.replace(/\{\{(\w+(?:\.\w+)*)\}\}/g, (_, path) => {
            return this.resolvePath(path, context);
        });
    }

    private resolvePath(path: string, context: TemplateContext): string {
        const parts = path.split('.');
        let value: any = context;

        for (const part of parts) {
            if (value === undefined || value === null) {
                return '';
            }
            value = value[part];
        }

        if (typeof value === 'object') {
            return JSON.stringify(value, null, 2);
        }

        return String(value ?? '');
    }
}

const template = `

Focus areas:
{{#each focusAreas}}

{{/each}}

Current file: {{file.path}}
`;

Skill 组合

依赖管理

// 推测的 Skill 组合实现
class SkillComposer {
    async compose(skills: Skill[], input: any): Promise<ComposedSkill> {
        // 构建依赖图
        const graph = this.buildDependencyGraph(skills);

        // 拓扑排序
        const order = this.topologicalSort(graph);

        // 创建组合 Skill
        return {
            id: `composed-${generateId()}`,
            skills: order,
            execute: async (context) => {
                const results: any[] = [];

                for (const skill of order) {
                    const result = await this.runSkill(skill, {
                        ...input,
                        previousResults: results
                    });
                    results.push(result);
                }

                return this.mergeResults(results);
            }
        };
    }

    private buildDependencyGraph(skills: Skill[]): DependencyGraph {
        const graph: DependencyGraph = new Map();

        for (const skill of skills) {
            const deps = skill.dependencies?.skills || [];
            graph.set(skill.id, deps);
        }

        return graph;
    }
}

Skill 设计原则

  • 单一职责：每个 Skill 只做一件事
  • 明确触发：定义清晰的触发条件
