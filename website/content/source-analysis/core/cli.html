// cli/index.ts (推测结构)

import { createCLI } from './cli';
import { loadConfig } from '../utils/config';
import { setupLogger } from '../utils/logger';
import { checkPermissions } from '../utils/permissions';

async function main(): Promise<void> {
    // 1. 全局错误处理
    process.on('uncaughtException', handleError);
    process.on('unhandledRejection', handleRejection);

    // 2. 加载配置
    const config = await loadConfig();
    
    // 3. 设置日志
    const logger = setupLogger(config.logLevel);
    
    // 4. 检查权限
    await checkPermissions(config.permissions);
    
    // 5. 创建 CLI 实例
    const cli = await createCLI(config);
    
    // 6. 解析参数并执行
    const args = parseArgs(process.argv.slice(2));
    await cli.run(args);
}

main().catch(exitOnError);

命令注册示例

// 推测的命令注册实现
class CommandRegistry {
    private commands: Map<string, Command> = new Map();
    private aliases: Map<string, string> = new Map();

    register(command: Command): void {
        this.commands.set(command.name, command);
        
        // 注册别名
        if (command.aliases) {
            for (const alias of command.aliases) {
                this.aliases.set(alias, command.name);
            }
        }
    }

    resolve(name: string): Command | undefined {
        // 先查找命令名
        const command = this.commands.get(name);
        if (command) return command;
        
        // 再查找别名
        const realName = this.aliases.get(name);
        return realName ? this.commands.get(realName) : undefined;
    }

    parse(input: string[]): ParsedCommand {
        const [name, ...args] = input;
        const command = this.resolve(name);
        
        if (!command) {
            throw new UnknownCommandError(name);
        }

        return {
            command,
            args: this.parseArgs(args, command.options)
        };
    }
}

交互式界面 (REPL)

REPL 架构

// 推测的 REPL 实现
import * as readline from 'readline';

class REPL {
    private rl: readline.Interface;
    private history: string[] = [];
    private completer: Completer;
    private highlighter: Highlighter;

    constructor(private cli: CLI) {
        this.rl = readline.createInterface({
            input: process.stdin,
            output: process.stdout,
            terminal: true,
            historySize: 1000,
            removeHistoryDuplicates: true
        });

        this.setupEventHandlers();
    }

    private setupEventHandlers(): void {
        // 处理输入
        this.rl.on('line', async (input) => {
            await this.handleInput(input);
        });

        // 处理关闭
        this.rl.on('close', () => {
            this.cleanup();
            process.exit(0);
        });

        // 处理中断 (Ctrl+C)
        process.on('SIGINT', () => {
            this.handleInterrupt();
        });
    }

    async start(): Promise<void> {
        this.showWelcome();
        this.rl.prompt();
    }

    private async handleInput(input: string): Promise<void> {
        const trimmed = input.trim();
        
        if (!trimmed) {
            this.rl.prompt();
            return;
        }

        // 添加到历史
        this.history.push(trimmed);

        try {
            // 检查是否是斜杠命令
            if (trimmed.startsWith('/')) {
                await this.handleSlashCommand(trimmed);
            } else {
                // 作为自然语言处理
                await this.handleNaturalLanguage(trimmed);
            }
        } catch (error) {
            this.showError(error);
        }

        this.rl.prompt();
    }

    private async handleSlashCommand(input: string): Promise<void> {
        const [command, ...args] = input.slice(1).split(/\s+/);
        
        switch (command) {
            case 'help':
                this.showHelp();
                break;
            case 'clear':
                this.clearScreen();
                break;
            case 'exit':
            case 'quit':
                this.rl.close();
                break;
            case 'config':
                await this.handleConfigCommand(args);
                break;
            case 'memory':
                await this.handleMemoryCommand(args);
                break;
            default:
                await this.cli.executeCommand(command, args);
        }
    }
}

自动补全实现

// 推测的自动补全实现
class Completer {
    constructor(
        private commandRegistry: CommandRegistry,
        private fileSystem: FileSystem
    ) {}

    async complete(input: string): Promise<[string[], string]> {
        // 斜杠命令补全
        if (input.startsWith('/')) {
            return this.completeCommand(input);
        }

        // 文件路径补全
        if (this.looksLikePath(input)) {
            return this.completePath(input);
        }

        // 上下文相关补全
        return this.completeContext(input);
    }

    private completeCommand(input: string): [string[], string] {
        const partial = input.slice(1);
        const commands = this.commandRegistry.getAll();
        
        const matches = commands
            .filter(cmd => cmd.name.startsWith(partial))
            .map(cmd => `/${cmd.name}`);

        return [matches, partial];
    }

    private async completePath(input: string): Promise<[string[], string]> {
        const lastSlash = input.lastIndexOf('/');
        const dir = lastSlash >= 0 ? input.slice(0, lastSlash) : '.';
        const prefix = lastSlash >= 0 ? input.slice(lastSlash + 1) : input;

        const files = await this.fileSystem.readdir(dir);
        const matches = files
            .filter(f => f.startsWith(prefix))
            .map(f => {
                const fullPath = dir === '.' ? f : `${dir}/${f}`;
                return fullPath;
            });

        return [matches, prefix];
    }
}

输出格式化

输出类型

// 推测的输出格式化实现
interface OutputFormatter {
    formatText(text: string): string;
    formatCode(code: string, language: string): string;
    formatTable(data: any[], columns: string[]): string;
    formatError(error: Error): string;
    formatProgress(current: number, total: number): string;
}

class TerminalFormatter implements OutputFormatter {
    private colors = {
        reset: '\x1b[0m',
        bold: '\x1b[1m',
        dim: '\x1b[2m',
        green: '\x1b[32m',
        yellow: '\x1b[33m',
        blue: '\x1b[34m',
        red: '\x1b[31m',
        cyan: '\x1b[36m'
    };

    formatText(text: string): string {
        // 处理 Markdown 格式
        return text
            .replace(/\*\*(.+?)\*\*/g, `${this.colors.bold}$1${this.colors.reset}`)
            .replace(/\*(.+?)\*/g, `${this.colors.cyan}$1${this.colors.reset}`)
            .replace(/`(.+?)`/g, `${this.colors.yellow}$1${this.colors.reset}`);
    }

    formatCode(code: string, language: string): string {
        const lines = code.split('\n');
        const maxLineNum = lines.length.toString().length;
        
        return lines.map((line, i) => {
            const lineNum = (i + 1).toString().padStart(maxLineNum);
            return `${this.colors.dim}${lineNum}│${this.colors.reset} ${line}`;
        }).join('\n');
    }

    formatTable(data: any[], columns: string[]): string {
        const widths = columns.map(col => 
            Math.max(col.length, ...data.map(row => String(row[col]).length))
        );

        const header = columns.map((col, i) => 
            this.colors.bold + col.padEnd(widths[i]) + this.colors.reset
        ).join(' │ ');

        const separator = widths.map(w => '─'.repeat(w)).join('─┼─');

        const rows = data.map(row => 
            columns.map((col, i) => String(row[col]).padEnd(widths[i])).join(' │ ')
        );

        return [header, separator, ...rows].join('\n');
    }

    formatProgress(current: number, total: number): string {
        const percent = Math.round((current / total) * 100);
        const filled = Math.round(percent / 2);
        const empty = 50 - filled;
        
        const bar = `${'█'.repeat(filled)}${'░'.repeat(empty)}`;
        return `${this.colors.cyan}${bar}${this.colors.reset} ${percent}%`;
    }
}

// 推测的会话管理实现
class SessionManager {
    private currentSession: Session | null = null;
    private sessions: Map<string, Session> = new Map();

    async createSession(options: SessionOptions): Promise<Session> {
        const session: Session = {
            id: generateId(),
            createdAt: Date.now(),
            messages: [],
            context: {
                files: new Map(),
                variables: new Map()
            },
            config: options.config
        };

        this.sessions.set(session.id, session);
        this.currentSession = session;

        // 持久化会话
        await this.persistSession(session);

        return session;
    }

    async loadSession(sessionId: string): Promise<Session> {
        const session = await this.readSessionFromDisk(sessionId);
        if (!session) {
            throw new SessionNotFoundError(sessionId);
        }

        this.sessions.set(session.id, session);
        this.currentSession = session;

        return session;
    }

    async saveSession(): Promise<void> {
        if (!this.currentSession) return;
        await this.persistSession(this.currentSession);
    }

class SignalHandler {
    private isShuttingDown = false;

    constructor(private sessionManager: SessionManager) {
        this.setupHandlers();
    }

    private setupHandlers(): void {
        // Ctrl+C
        process.on('SIGINT', () => this.handleInterrupt());

        // 终端关闭
        process.on('SIGHUP', () => this.handleHangup());

        // 正常退出
        process.on('SIGTERM', () => this.handleTerminate());

        // 进程退出前
        process.on('beforeExit', () => this.cleanup());
    }

    private async handleInterrupt(): Promise<void> {
        if (this.isShuttingDown) {
            // 第二次 Ctrl+C，强制退出
            process.exit(1);
        }

        this.isShuttingDown = true;
        console.log('\n正在保存会话...');

        try {
            await this.sessionManager.saveSession();
            console.log('会话已保存。再见！');
            process.exit(0);
        } catch (error) {
            console.error('保存会话失败:', error);
            process.exit(1);
        }
    }

    private async cleanup(): Promise<void> {
        // 清理临时文件
        // 关闭连接
        // 释放资源
    }
}

设计模式总结

命令模式

将每个命令封装为独立对象，支持撤销、重做和历史记录。

责任链模式

中间件链处理请求，支持日志、认证、错误处理等横切关注点。

观察者模式

事件系统允许各模块订阅和响应 CLI 事件。

工厂模式

根据配置创建不同类型的输出格式化器和处理器。
