上下文管理概述

上下文管理是 Claude Code 的核心能力之一，负责在有限的 Token 预算内，最大化有效信息的传递。本节深入分析上下文窗口管理、消息压缩和记忆检索机制。

Token 计算器

// 推测的 Token 计算实现
class TokenCounter {
    private tokenizer: Tokenizer;

    constructor() {
        // 使用 Claude 的 tokenizer
        this.tokenizer = new ClaudeTokenizer();
    }

    count(text: string): number {
        return this.tokenizer.encode(text).length;
    }

    countMessages(messages: Message[]): number {
        let total = 0;

        for (const message of messages) {
            // 消息格式开销
            total += 4; // role + content 标签

            for (const block of message.content) {
                if (block.type === 'text') {
                    total += this.count(block.text);
                } else if (block.type === 'tool_use') {
                    total += this.count(block.name);
                    total += this.count(JSON.stringify(block.input));
                    total += 4; // 工具调用格式开销
                } else if (block.type === 'tool_result') {
                    total += this.count(block.content);
                    total += 4;
                }
            }
        }

        return total;
    }

    countTools(tools: ToolDefinition[]): number {
        let total = 0;

        for (const tool of tools) {
            total += this.count(tool.name);
            total += this.count(tool.description);
            total += this.count(JSON.stringify(tool.input_schema));
            total += 10; // 工具定义格式开销
        }

        return total;
    }
}

预算分配器

// 推测的预算分配实现
class BudgetAllocator {
    private readonly TOTAL_BUDGET = 200000;
    private readonly RESERVED_FOR_RESPONSE = 35000;

    allocate(context: ContextRequest): BudgetAllocation {
        const available = this.TOTAL_BUDGET - this.RESERVED_FOR_RESPONSE;

        const allocation: BudgetAllocation = {
            system: 0,
            memory: 0,
            files: 0,
            history: 0
        };

        // 1. 系统提示词 (必需)
        allocation.system = Math.min(
            context.systemPromptTokens,
            available * 0.1
        );

        const remaining = available - allocation.system;

        // 2. 记忆上下文 (高优先级)
        allocation.memory = Math.min(
            context.memoryTokens,
            remaining * 0.1
        );

        const afterMemory = remaining - allocation.memory;

        // 3. 文件上下文 (动态调整)
        const fileBudget = afterMemory * 0.5;
        allocation.files = this.allocateFiles(
            context.requestedFiles,
            fileBudget
        );

        // 4. 历史记录 (剩余空间)
        allocation.history = afterMemory - allocation.files;

        return allocation;
    }

    private allocateFiles(
        files: FileRequest[], 
        budget: number
    ): number {
        // 按重要性排序
        const sorted = files.sort((a, b) => b.priority - a.priority);

        let used = 0;
        for (const file of sorted) {
            if (used + file.estimatedTokens <= budget) {
                used += file.estimatedTokens;
                file.included = true;
            } else {
                file.included = false;
            }
        }

        return used;
    }
}

消息压缩策略

压缩触发条件

// 推测的压缩触发机制
class CompressionTrigger {
    private threshold = 0.8; // 80% 使用率触发压缩

    shouldCompress(currentTokens: number, maxTokens: number): boolean {
        const usage = currentTokens / maxTokens;
        return usage >= this.threshold;
    }

    getCompressionTarget(currentTokens: number, maxTokens: number): number {
        // 压缩到 60% 使用率
        return Math.floor(maxTokens * 0.6);
    }
}

压缩算法

// 推测的压缩实现
class MessageCompressor {
    constructor(private llmClient: LLMClient) {}

    async compress(
        messages: Message[], 
        targetTokens: number
    ): Promise<CompressedHistory> {
        const currentTokens = this.countTokens(messages);
        
        if (currentTokens <= targetTokens) {
            return { messages, summary: null };
        }

        // 策略 1: 保留最近消息，压缩早期消息
        const recentCount = 5;
        const recentMessages = messages.slice(-recentCount);
        const oldMessages = messages.slice(0, -recentCount);

        // 策略 2: 生成摘要
        const summary = await this.generateSummary(oldMessages);

        // 策略 3: 提取关键信息
        const keyDecisions = this.extractKeyDecisions(oldMessages);
        const toolCalls = this.extractToolCalls(oldMessages);

        // 构建压缩后的历史
        const compressedMessages: Message[] = [
            {
                role: 'user',
                content: [{
                    type: 'text',
                    text: `[历史摘要]\n${summary}\n\n[关键决策]\n${keyDecisions.join('\n')}`
                }]
            },
            ...recentMessages
        ];

        return {
            messages: compressedMessages,
            summary,
            keyDecisions,
            toolCalls,
            compressionRatio: currentTokens / this.countTokens(compressedMessages)
        };
    }

    private async generateSummary(messages: Message[]): Promise<string> {
        const content = messages.map(m => 
            `${m.role}: ${this.extractText(m)}`
        ).join('\n\n');

        const response = await this.llmClient.chat({
            messages: [{
                role: 'user',
                content: `请总结以下对话的关键内容，保留重要决策和上下文：\n\n${content}`
            }],
            maxTokens: 1000
        });

        return response.content;
    }

    private extractKeyDecisions(messages: Message[]): string[] {
        const decisions: string[] = [];
        const patterns = [
            /决定[:：](.+)/,
            /选择[:：](.+)/,
            /使用(.+)方案/,
            /最终答案[:：](.+)/
        ];

        for (const message of messages) {
            const text = this.extractText(message);
            for (const pattern of patterns) {
                const match = text.match(pattern);
                if (match) {
                    decisions.push(match[1].trim());
                }
            }
        }

        return decisions;
    }
}

滑动窗口实现

// 推测的滑动窗口实现
class SlidingWindow {
    private windowSize: number;
    private overlap: number;

    constructor(windowSize: number, overlap: number = 2) {
        this.windowSize = windowSize;
        this.overlap = overlap;
    }

    apply(messages: Message[]): Message[] {
        if (messages.length <= this.windowSize) {
            return messages;
        }

        // 保留第一条消息 (通常包含初始请求)
        const first = messages[0];

        // 取最近的窗口大小消息
        const recent = messages.slice(-(this.windowSize - 1));

        // 添加重叠上下文
        const overlapMessages = messages.slice(
            -(this.windowSize + this.overlap),
            -this.windowSize
        );

        return [
            first,
            ...this.createContextBridge(overlapMessages),
            ...recent
        ];
    }

    private createContextBridge(messages: Message[]): Message[] {
        if (messages.length === 0) return [];

        // 创建一个桥接消息，总结被省略的上下文
        const summary = this.summarizeMessages(messages);
        
        return [{
            role: 'user',
            content: [{
                type: 'text',
                text: `[上下文桥接] ${summary}`
            }]
        }];
    }
}

智能文件加载

// 推测的文件上下文管理
class FileContextManager {
    private fileCache: Map<string, CachedFile> = new Map();
    private tokenBudget: number;

    async loadFile(path: string, priority: number): Promise<FileContext> {
        // 检查缓存
        const cached = this.fileCache.get(path);
        if (cached && !this.isStale(cached)) {
            return cached.context;
        }

        // 读取文件
        const content = await fs.readFile(path, 'utf-8');
        const stats = await fs.stat(path);
        const language = this.detectLanguage(path);

        const context: FileContext = {
            path,
            content,
            language,
            lastModified: stats.mtimeMs,
            tokens: this.countTokens(content),
            priority
        };

        // 更新缓存
        this.fileCache.set(path, {
            context,
            cachedAt: Date.now()
        });

        return context;
    }

    async selectRelevantFiles(
        query: string, 
        projectFiles: string[], 
        budget: number
    ): Promise<FileContext[]> {
        // 1. 分析查询意图
        const intent = await this.analyzeQuery(query);

        // 2. 计算文件相关性分数
        const scores = await Promise.all(
            projectFiles.map(path => this.calculateRelevance(path, intent))
        );

        // 3. 按相关性排序
        const ranked = projectFiles
            .map((path, i) => ({ path, score: scores[i] }))
            .sort((a, b) => b.score - a.score);

        // 4. 在预算内选择文件
        const selected: FileContext[] = [];
        let usedTokens = 0;

        for (const { path } of ranked) {
            const context = await this.loadFile(path, 1);
            
            if (usedTokens + context.tokens <= budget) {
                selected.push(context);
                usedTokens += context.tokens;
            } else {
                // 尝试加载部分内容
                const partial = await this.loadPartialFile(path, budget - usedTokens);
                if (partial) {
                    selected.push(partial);
                }
                break;
            }
        }

        return selected;
    }

    private async calculateRelevance(
        path: string, 
        intent: QueryIntent
    ): Promise<number> {
        let score = 0;

        // 文件名匹配
        if (intent.mentionedFiles?.some(f => path.includes(f))) {
            score += 0.5;
        }

        // 文件类型匹配
        if (intent.fileTypes?.includes(this.getExtension(path))) {
            score += 0.2;
        }

        // 最近修改
        const stats = await fs.stat(path);
        const age = Date.now() - stats.mtimeMs;
        if (age < 3600000) { // 1小时内
            score += 0.3;
        }

        return Math.min(score, 1);
    }
}

记忆检索系统

记忆索引

// 推测的记忆检索实现
class MemoryIndex {
    private index: VectorIndex;
    private keywordIndex: Map<string, Set<string>>;

    async indexMemory(memory: Memory): Promise<void> {
        // 向量索引
        const embedding = await this.getEmbedding(memory.content);
        await this.index.add(memory.id, embedding, {
            type: memory.type,
            timestamp: memory.timestamp,
            tags: memory.tags
        });

        // 关键词索引
        const keywords = this.extractKeywords(memory.content);
        for (const keyword of keywords) {
            if (!this.keywordIndex.has(keyword)) {
                this.keywordIndex.set(keyword, new Set());
            }
            this.keywordIndex.get(keyword)!.add(memory.id);
        }
    }

    async search(query: string, limit: number = 10): Promise<Memory[]> {
        // 混合检索：向量 + 关键词

        // 1. 向量检索
        const queryEmbedding = await this.getEmbedding(query);
        const vectorResults = await this.index.search(queryEmbedding, limit * 2);

        // 2. 关键词检索
        const keywords = this.extractKeywords(query);
        const keywordResults = new Set<string>();
        for (const keyword of keywords) {
            const matches = this.keywordIndex.get(keyword);
            if (matches) {
                matches.forEach(id => keywordResults.add(id));
            }
        }

        // 3. 合并结果
        const combined = this.mergeResults(
            vectorResults,
            Array.from(keywordResults),
            0.7 // 向量权重
        );

        // 4. 返回 Top N
        return combined.slice(0, limit);
    }

    private mergeResults(
        vectorResults: SearchResult[],
        keywordIds: string[],
        vectorWeight: number
    ): Memory[] {
        const scores = new Map<string, number>();

        // 向量分数
        for (const result of vectorResults) {
            scores.set(result.id, result.score * vectorWeight);
        }

        // 关键词分数
        const keywordWeight = 1 - vectorWeight;
        for (const id of keywordIds) {
            const current = scores.get(id) || 0;
            scores.set(id, current + keywordWeight);
        }

        // 排序并返回
        return Array.from(scores.entries())
            .sort((a, b) => b[1] - a[1])
            .map(([id]) => this.getMemory(id));
    }
}
