任务规划器 (Planner)

规划流程

// 推测的任务规划实现
interface Plan {
    id: string;
    goal: string;
    steps: PlanStep[];
    dependencies: Map<string, string[]>;
    estimatedComplexity: 'low' | 'medium' | 'high';
}

interface PlanStep {
    id: string;
    description: string;
    action: Action;
    status: 'pending' | 'running' | 'completed' | 'failed';
    dependencies: string[];
    retryCount: number;
}

class Planner {
    constructor(
        private llmClient: LLMClient,
        private toolRegistry: ToolRegistry
    ) {}

    async plan(userInput: string, context: Context): Promise<Plan> {
        // 1. 理解用户意图
        const intent = await this.understandIntent(userInput, context);
        
        // 2. 评估任务复杂度
        const complexity = this.assessComplexity(intent, context);
        
        // 3. 生成执行计划
        if (complexity === 'low') {
            return this.createSimplePlan(intent);
        } else {
            return this.createComplexPlan(intent, context);
        }
    }

    private async understandIntent(
        input: string, 
        context: Context
    ): Promise<Intent> {
        const response = await this.llmClient.chat({
            messages: [
                { role: 'system', content: INTENT_ANALYSIS_PROMPT },
                { role: 'user', content: input }
            ],
            context: {
                files: context.files,
                recentActions: context.recentActions
            }
        });

        return this.parseIntentResponse(response);
    }

    private assessComplexity(intent: Intent, context: Context): Complexity {
        // 简单启发式规则
        let score = 0;

        // 涉及多个文件
        if (intent.affectedFiles?.length > 3) score += 2;
        
        // 需要外部工具
        if (intent.requiresExternalTools) score += 1;
        
        // 需要代码生成
        if (intent.requiresCodeGeneration) score += 1;
        
        // 涉及多个步骤
        if (intent.estimatedSteps > 5) score += 2;

        if (score <= 2) return 'low';
        if (score <= 5) return 'medium';
        return 'high';
    }

    private async createComplexPlan(
        intent: Intent, 
        context: Context
    ): Promise<Plan> {
        const response = await this.llmClient.chat({
            messages: [
                { role: 'system', content: PLANNING_PROMPT },
                { role: 'user', content: this.formatPlanningRequest(intent) }
            ],
            tools: this.toolRegistry.getToolDefinitions()
        });

        return this.parsePlanResponse(response);
    }
}

执行器 (Executor)

执行流程

// 推测的执行器实现
class Executor {
    private maxIterations = 20;
    private iterationCount = 0;

    constructor(
        private toolRegistry: ToolRegistry,
        private stateManager: StateManager,
        private eventBus: EventBus
    ) {}

    async execute(plan: Plan, context: Context): Promise<ExecutionResult> {
        const result: ExecutionResult = {
            success: false,
            steps: [],
            output: null
        };

        while (this.iterationCount < this.maxIterations) {
            // 获取下一个可执行的步骤
            const nextStep = this.getNextStep(plan);
            
            if (!nextStep) {
                // 所有步骤完成
                result.success = true;
                break;
            }

            try {
                // 执行步骤
                const stepResult = await this.executeStep(nextStep, context);
                result.steps.push(stepResult);

                // 更新计划状态
                plan.steps.find(s => s.id === nextStep.id)!.status = 'completed';

                // 检查是否需要调整计划
                if (stepResult.requiresReplanning) {
                    plan = await this.replan(plan, context);
                }

            } catch (error) {
                // 处理执行错误
                const handled = await this.handleError(error, nextStep, plan);
                
                if (!handled) {
                    result.success = false;
                    result.error = error;
                    break;
                }
            }

            this.iterationCount++;
        }

        return result;
    }

    private async executeStep(
        step: PlanStep, 
        context: Context
    ): Promise<StepResult> {
        this.eventBus.emit('step:start', { step });

        const startTime = Date.now();

        try {
            // 根据动作类型执行
            let result: any;

            switch (step.action.type) {
                case 'tool_call':
                    result = await this.executeToolCall(step.action, context);
                    break;
                case 'llm_call':
                    result = await this.executeLLMCall(step.action, context);
                    break;
                case 'subagent':
                    result = await this.executeSubagent(step.action, context);
                    break;
                default:
                    throw new UnknownActionError(step.action.type);
            }

            const duration = Date.now() - startTime;

            this.eventBus.emit('step:complete', { step, result, duration });

            return {
                stepId: step.id,
                success: true,
                output: result,
                duration
            };

        } catch (error) {
            this.eventBus.emit('step:error', { step, error });
            throw error;
        }
    }

    private async executeToolCall(
        action: ToolAction, 
        context: Context
    ): Promise<any> {
        const tool = this.toolRegistry.get(action.toolName);
        
        if (!tool) {
            throw new ToolNotFoundError(action.toolName);
        }

        // 权限检查
        const permitted = await this.checkPermission(tool, action.params);
        if (!permitted) {
            throw new PermissionDeniedError(tool.name);
        }

        // 执行工具
        return tool.execute(action.params, context);
    }
}

迭代执行模式

// ReAct (Reasoning + Acting) 模式
class ReActExecutor {
    async execute(query: string, context: Context): Promise<string> {
        let thought: string;
        let action: Action | null = null;
        let observation: string;
        
        const trajectory: TrajectoryEntry[] = [];

        while (true) {
            // Thought: 思考下一步
            thought = await this.think(query, trajectory, context);
            trajectory.push({ type: 'thought', content: thought });

            // 检查是否可以得出结论
            if (this.isConclusion(thought)) {
                return this.extractAnswer(thought);
            }

            // Action: 决定执行什么动作
            action = await this.decideAction(thought, context);
            trajectory.push({ type: 'action', content: action });

            // Observation: 执行动作并观察结果
            observation = await this.observe(action, context);
            trajectory.push({ type: 'observation', content: observation });

            // 检查是否达到最大迭代次数
            if (trajectory.length >= this.maxSteps * 3) {
                return this.generatePartialAnswer(trajectory);
            }
        }
    }

    private async think(
        query: string, 
        trajectory: TrajectoryEntry[], 
        context: Context
    ): Promise<string> {
        const prompt = this.buildThoughtPrompt(query, trajectory);
        
        const response = await this.llmClient.chat({
            messages: [
                { role: 'system', content: REACT_SYSTEM_PROMPT },
                { role: 'user', content: prompt }
            ]
        });

        return response.content;
    }

    private isConclusion(thought: string): boolean {
        return thought.includes('Final Answer:') || 
               thought.includes('最终答案:');
    }
}

// 推测的状态管理实现
interface AgentState {
    // 会话信息
    sessionId: string;
    createdAt: number;
    updatedAt: number;

    // 对话历史
    messages: Message[];

    // 文件状态
    files: Map<string, FileState>;

    // 执行状态
    execution: {
        currentPlan: Plan | null;
        completedSteps: string[];
        pendingSteps: string[];
    };

    // 上下文变量
    variables: Map<string, any>;

    // 元数据
    metadata: {
        totalTokens: number;
        toolCalls: number;
        errors: ErrorRecord[];
    };
}

interface FileState {
    path: string;
    lastModified: number;
    checksum: string;
    content?: string;
    language?: string;
}

class StateManager {
    private state: AgentState;
    private persistence: Persistence;

    constructor(sessionId: string) {
        this.state = this.createInitialState(sessionId);
        this.persistence = new FilePersistence();
    }

    updateState(updates: Partial<AgentState>): void {
        this.state = {
            ...this.state,
            ...updates,
            updatedAt: Date.now()
        };

        // 触发状态变更事件
        this.emitStateChange(updates);
    }

    addMessage(message: Message): void {
        this.state.messages.push(message);
        this.updateState({});
    }

    updateFile(path: string, state: Partial<FileState>): void {
        const existing = this.state.files.get(path) || {
            path,
            lastModified: 0,
            checksum: ''
        };

        this.state.files.set(path, {
            ...existing,
            ...state,
            lastModified: Date.now()
        });
    }

    async save(): Promise<void> {
        await this.persistence.save(this.state);
    }

    async restore(sessionId: string): Promise<void> {
        this.state = await this.persistence.load(sessionId);
    }
}

错误处理与恢复

// 推测的错误处理实现
class ErrorHandler {
    async handle(
        error: Error, 
        step: PlanStep, 
        plan: Plan, 
        context: Context
    ): Promise<boolean> {
        // 记录错误
        context.metadata.errors.push({
            stepId: step.id,
            error: error.message,
            timestamp: Date.now()
        });

        // 根据错误类型决定处理策略
        if (error instanceof ToolExecutionError) {
            return this.handleToolError(error, step, plan);
        }

        if (error instanceof PermissionDeniedError) {
            return this.handlePermissionError(error, step, plan);
        }

        if (error instanceof LLMRateLimitError) {
            return this.handleRateLimitError(error, step, plan);
        }

        if (error instanceof TimeoutError) {
            return this.handleTimeoutError(error, step, plan);
        }

        // 未知错误，尝试重新规划
        return this.handleUnknownError(error, step, plan, context);
    }

    private async handleToolError(
        error: ToolExecutionError,
        step: PlanStep,
        plan: Plan
    ): Promise<boolean> {
        // 检查重试次数
        if (step.retryCount < 3) {
            step.retryCount++;
            step.status = 'pending';
            return true; // 可以继续
        }

        // 标记步骤失败
        step.status = 'failed';
        return false;
    }

    private async handleRateLimitError(
        error: LLMRateLimitError,
        step: PlanStep,
        plan: Plan
    ): Promise<boolean> {
        // 等待后重试
        const waitTime = error.retryAfter || 60000;
        await this.sleep(waitTime);
        
        step.status = 'pending';
        return true;
    }

    private async handleUnknownError(
        error: Error,
        step: PlanStep,
        plan: Plan,
        context: Context
    ): Promise<boolean> {
        // 尝试使用 LLM 分析错误并生成替代方案
        const alternative = await this.generateAlternative(error, step, context);
        
        if (alternative) {
            // 替换失败的步骤
            const index = plan.steps.findIndex(s => s.id === step.id);
            plan.steps[index] = alternative;
            return true;
        }

        return false;
    }
}
