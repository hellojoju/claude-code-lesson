插件上下文

上下文对象

interface PluginContext {
  // 基本信息
  extensionPath: string;      // 插件路径
  globalState: State;         // 全局状态
  workspaceState: State;      // 工作区状态
  
  // 注册方法
  registerCommand: (name, handler) => void;
  registerSkill: (config) => void;
  registerHook: (event, handler) => void;
  
  // 工具方法
  showInformationMessage: (message) => void;
  showErrorMessage: (message) => void;
  showInputBox: (options) => Promise;
  
  // 订阅管理
  subscriptions: Disposable[];
}

使用上下文

export async function activate(context: PluginContext) {
  // 使用状态存储
  const count = context.globalState.get('count', 0);
  context.globalState.set('count', count + 1);
  
  // 注册命令
  const disposable = context.registerCommand('myPlugin.hello', () => {
    context.showInformationMessage('Hello!');
  });
  
  // 添加到订阅
  context.subscriptions.push(disposable);
}

插件隔离

隔离机制

  • 进程隔离：插件运行在独立进程
  • API 隔离：只能访问授权的 API
  • 状态隔离：插件状态相互独立
  • 错误隔离：一个插件崩溃不影响其他

权限声明

{
  "claudeCode": {
    "permissions": [
      "files:read",
      "files:write",
      "commands:execute",
      "network:fetch"
    ]
  }
}
