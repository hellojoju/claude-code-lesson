案例概述

本案例将手把手教你开发一个外部服务集成插件，实现与 GitHub、Jira、Slack 等服务的无缝集成。这类插件可以显著提升开发效率，让 Claude Code 成为你的开发中心。

学习目标

  • 理解外部服务集成的设计模式

  • 学习 API 调用的最佳实践

第二步：实现 GitHub 集成

2.1 GitHub 客户端封装

// src/services/github/client.ts
import { Octokit } from '@octokit/rest';
import { IssueTracker, CodeRepository } from '../base';

export class GitHubClient implements IssueTracker, CodeRepository {
    private octokit: Octokit;
    private config: GitHubConfig;
    private cache: CacheManager;

    constructor(config: GitHubConfig, cache: CacheManager) {
        this.config = config;
        this.cache = cache;
        this.octokit = new Octokit({
            auth: this.getAuthToken(),
            baseUrl: config.baseUrl || 'https://api.github.com'
        });
    }

    async createIssue(params: CreateIssueParams): Promise<Issue> {
        const { owner, repo } = this.parseRepo(params.repository);
        
        const response = await this.octokit.issues.create({
            owner,
            repo,
            title: params.title,
            body: params.body,
            labels: params.labels,
            assignees: params.assignees
        });

        return this.mapToIssue(response.data);
    }

    async listIssues(filter: IssueFilter): Promise<Issue[]> {
        const cacheKey = `issues:${JSON.stringify(filter)}`;
        const cached = await this.cache.get(cacheKey);
        if (cached) return cached;

        const { owner, repo } = this.parseRepo(filter.repository);
        
        const response = await this.octokit.issues.listForRepo({
            owner,
            repo,
            state: filter.state || 'open',
            labels: filter.labels?.join(','),
            assignee: filter.assignee,
            per_page: filter.limit || 30
        });

        const issues = response.data.map(this.mapToIssue);
        await this.cache.set(cacheKey, issues, 300);
        
        return issues;
    }

    async createPR(params: CreatePRParams): Promise<PullRequest> {
        const { owner, repo } = this.parseRepo(params.repository);
        
        const response = await this.octokit.pulls.create({
            owner,
            repo,
            title: params.title,
            head: params.headBranch,
            base: params.baseBranch,
            body: params.body,
            draft: params.draft
        });

        return this.mapToPR(response.data);
    }

    private parseRepo(repository: string): { owner: string; repo: string } {
        const [owner, repo] = repository.split('/');
        return { owner, repo };
    }
}

2.2 OAuth 认证流程

// src/services/github/oauth.ts
export class GitHubOAuth {
    private clientId: string;
    private clientSecret: string;
    private redirectUri: string;

    constructor(config: OAuthConfig) {
        this.clientId = config.clientId;
        this.clientSecret = config.clientSecret;
        this.redirectUri = config.redirectUri;
    }

    getAuthorizationUrl(scopes: string[] = ['repo', 'user']): string {
        const state = crypto.randomBytes(16).toString('hex');
        const params = new URLSearchParams({
            client_id: this.clientId,
            redirect_uri: this.redirectUri,
            scope: scopes.join(' '),
            state: state
        });

        return `https://github.com/login/oauth/authorize?${params}`;
    }

    async exchangeCodeForToken(code: string): Promise<TokenResponse> {
        const response = await fetch('https://github.com/login/oauth/access_token', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: JSON.stringify({
                client_id: this.clientId,
                client_secret: this.clientSecret,
                code: code,
                redirect_uri: this.redirectUri
            })
        });

        return response.json();
    }

    async refreshToken(refreshToken: string): Promise<TokenResponse> {
        // GitHub OAuth tokens don't expire by default
        // Implement if using GitHub Apps
        throw new Error('Token refresh not supported for GitHub OAuth');
    }
}

第三步：实现 Slack 集成

3.1 Slack 客户端

// src/services/slack/client.ts
import { WebClient } from '@slack/web-api';
import { NotificationService } from '../base';

export class SlackClient implements NotificationService {
    private client: WebClient;
    private defaultChannel: string;

    constructor(config: SlackConfig) {
        this.client = new WebClient(config.botToken);
        this.defaultChannel = config.defaultChannel;
    }

    async send(message: NotificationMessage): Promise<void> {
        await this.client.chat.postMessage({
            channel: this.defaultChannel,
            text: message.text,
            blocks: this.buildBlocks(message),
            attachments: message.attachments
        });
    }

    async sendToChannel(channel: string, message: NotificationMessage): Promise<void> {
        await this.client.chat.postMessage({
            channel,
            text: message.text,
            blocks: this.buildBlocks(message)
        });
    }

    private buildBlocks(message: NotificationMessage): any[] {
        const blocks: any[] = [];

        if (message.title) {
            blocks.push({
                type: 'header',
                text: {
                    type: 'plain_text',
                    text: message.title
                }
            });
        }

        if (message.text) {
            blocks.push({
                type: 'section',
                text: {
                    type: 'mrkdwn',
                    text: message.text
                }
            });
        }

        if (message.fields) {
            blocks.push({
                type: 'section',
                fields: message.fields.map(field => ({
                    type: 'mrkdwn',
                    text: `*${field.label}*\n${field.value}`
                }))
            });
        }

        if (message.actions) {
            blocks.push({
                type: 'actions',
                elements: message.actions.map(action => ({
                    type: 'button',
                    text: { type: 'plain_text', text: action.label },
                    url: action.url
                }))
            });
        }

        return blocks;
    }
}

第四步：实现 Jira 集成

4.1 Jira 客户端

// src/services/jira/client.ts
export class JiraClient implements IssueTracker {
    private baseUrl: string;
    private auth: { email: string; token: string };

    constructor(config: JiraConfig) {
        this.baseUrl = config.baseUrl;
        this.auth = {
            email: config.email,
            token: config.apiToken
        };
    }

    private async request<T>(method: string, path: string, body?: any): Promise<T> {
        const response = await fetch(`${this.baseUrl}/rest/api/3${path}`, {
            method,
            headers: {
                'Authorization': `Basic ${Buffer.from(
                    `${this.auth.email}:${this.auth.token}`
                ).toString('base64')}`,
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: body ? JSON.stringify(body) : undefined
        });

        if (!response.ok) {
            throw new JiraError(await response.text(), response.status);
        }

        return response.json();
    }

    async createIssue(params: CreateIssueParams): Promise<Issue> {
        const response = await this.request<JiraIssue>('POST', '/issue', {
            fields: {
                project: { key: params.project },
                summary: params.title,
                description: this.toADF(params.body),
                issuetype: { name: params.type || 'Task' },
                labels: params.labels,
                assignee: params.assignee ? { accountId: params.assignee } : undefined
            }
        });

        return this.mapToIssue(response);
    }

    async addComment(issueId: string, comment: string): Promise<void> {
        await this.request('POST', `/issue/${issueId}/comment`, {
            body: this.toADF(comment)
        });
    }

    private toADF(text: string): any {
        return {
            type: 'doc',
            version: 1,
            content: [{
                type: 'paragraph',
                content: [{ type: 'text', text }]
            }]
        };
    }
}

第五步：集成管理器

5.1 统一管理接口

// src/integration/manager.ts
export class IntegrationManager {
    private services: Map<string, any> = new Map();
    private config: IntegrationConfig;

    constructor(config: IntegrationConfig) {
        this.config = config;
        this.initializeServices();
    }

    private initializeServices(): void {
        const { services, cache } = this.config;
        const cacheManager = new CacheManager(cache);

        if (services.github) {
            this.services.set('github', new GitHubClient(services.github, cacheManager));
        }

        if (services.gitlab) {
            this.services.set('gitlab', new GitLabClient(services.gitlab, cacheManager));
        }

        if (services.jira) {
            this.services.set('jira', new JiraClient(services.jira));
        }

        if (services.slack) {
            this.services.set('slack', new SlackClient(services.slack));
        }

        if (services.discord) {
            this.services.set('discord', new DiscordClient(services.discord));
        }
    }

    getIssueTracker(): IssueTracker {
        const name = this.config.defaults.issueTracker;
        return this.services.get(name);
    }

    getNotificationService(): NotificationService {
        const name = this.config.defaults.notifications;
        return this.services.get(name);
    }

    getCodeRepository(): CodeRepository {
        const name = this.config.defaults.repository;
        return this.services.get(name);
    }

    getService<T>(name: string): T {
        return this.services.get(name);
    }
}

5.2 插件命令注册

// src/commands.ts
export function registerIntegrationCommands(
    plugin: ClaudeCodePlugin,
    manager: IntegrationManager
): void {
    plugin.registerCommand('issue', {
        description: '管理 Issue',
        subCommands: {
            create: {
                description: '创建新 Issue',
                execute: async (args) => {
                    const tracker = manager.getIssueTracker();
                    const issue = await tracker.createIssue({
                        title: args.title,
                        body: args.body,
                        repository: args.repo,
                        labels: args.labels?.split(',')
                    });
                    return `Issue #${issue.number} 已创建: ${issue.url}`;
                }
            },
            list: {
                description: '列出 Issues',
                execute: async (args) => {
                    const tracker = manager.getIssueTracker();
                    const issues = await tracker.listIssues({
                        repository: args.repo,
                        state: args.state || 'open',
                        assignee: args.assignee
                    });
                    return issues.map(i => `#${i.number} ${i.title}`).join('\n');
                }
            }
        }
    });

    plugin.registerCommand('notify', {
        description: '发送通知',
        execute: async (args) => {
            const notifier = manager.getNotificationService();
            await notifier.send({
                title: args.title,
                text: args.message,
                channel: args.channel
            });
            return '通知已发送';
        }
    });

    plugin.registerCommand('pr', {
        description: '管理 Pull Request',
        subCommands: {
            create: {
                description: '创建 PR',
                execute: async (args) => {
                    const repo = manager.getCodeRepository();
                    const pr = await repo.createPR({
                        title: args.title,
                        body: args.body,
                        headBranch: args.head,
                        baseBranch: args.base || 'main',
                        repository: args.repo
                    });
                    return `PR #${pr.number} 已创建: ${pr.url}`;
                }
            }
        }
    });
}

第六步：错误处理与重试

// src/utils/retry.ts
export class RetryManager {
    static async withRetry<T>(
        operation: () => Promise<T>,
        options: RetryOptions = {}
    ): Promise<T> {
        const {
            maxAttempts = 3,
            backoff = 'exponential',
            initialDelay = 1000,
            maxDelay = 30000,
            retryOn = [408, 429, 500, 502, 503, 504]
        } = options;

        let lastError: Error;
        let delay = initialDelay;

        for (let attempt = 1; attempt <= maxAttempts; attempt++) {
            try {
                return await operation();
            } catch (error) {
                lastError = error;

                if (!this.shouldRetry(error, retryOn)) {
                    throw error;
                }

                if (attempt === maxAttempts) {
                    throw new MaxRetriesExceededError(
                        `操作失败，已重试 ${maxAttempts} 次`,
                        error
                    );
                }

                await this.sleep(delay);

                if (backoff === 'exponential') {
                    delay = Math.min(delay * 2, maxDelay);
                }
            }
        }

        throw lastError;
    }

    private static shouldRetry(error: any, retryOn: number[]): boolean {
        if (error.status && retryOn.includes(error.status)) {
            return true;
        }
        if (error.code === 'ECONNRESET' || error.code === 'ETIMEDOUT') {
            return true;
        }
        return false;
    }

    private static sleep(ms: number): Promise<void> {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

最佳实践总结

安全考虑

  • 使用环境变量存储敏感凭证
  • 实现 OAuth 流程而非硬编码 token
  • 定期轮换 API 密钥
  • 限制 token 的权限范围

性能优化

  • 实现请求缓存减少 API 调用
  • 使用批量操作减少请求次数
  • 实现请求限流避免触发 API 限制
  • 异步处理耗时操作

错误处理

  • 实现重试机制处理临时故障
  • 提供有意义的错误信息
  • 记录详细的调试日志
  • 优雅降级而非完全失败
