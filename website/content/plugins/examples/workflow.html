工作流插件案例：自动化开发流程

可靠性: A 来源: Claude Code 官方文档 + 社区最佳实践

案例概述

本案例将手把手教你开发一个工作流自动化插件，实现从代码编写到部署的全流程自动化。这类插件可以将繁琐的手动操作转变为一条命令完成的自动化流程。

学习目标

  • 理解工作流自动化的设计理念
  • 掌握状态机和工作流引擎的实现
  • 学习任务编排和依赖管理
  • 实现可配置的工作流模板

export interface Workflow {
    id: string;
    name: string;
    description: string;
    version: string;
    triggers: Trigger[];
    steps: WorkflowStep[];
    onError: ErrorHandler;
}

export interface WorkflowStep {
    id: string;
    name: string;
    type: StepType;
    action: Action;
    condition?: Condition;
    retryPolicy?: RetryPolicy;
    timeout?: number;
    dependencies?: string[];
}

export type StepType = 
    | 'task'           // 执行任务
    | 'decision'       // 条件分支
    | 'parallel'       // 并行执行
    | 'subworkflow';   // 嵌套工作流

export interface Action {
    type: string;
    params: Record<string, any>;
    output?: string;  // 输出变量名
}

export interface WorkflowContext {
    workflowId: string;
    executionId: string;
    variables: Map<string, any>;
    stepResults: Map<string, StepResult>;
    currentStep: string;
    status: ExecutionStatus;
}

export type ExecutionStatus = 
    | 'pending'
    | 'running'
    | 'paused'
    | 'completed'
    | 'failed'
    | 'cancelled';

1.2 工作流引擎实现

// src/workflow/engine.ts
export class WorkflowEngine {
    private actionRegistry: Map<string, ActionHandler>;
    private stateStore: StateStore;
    private eventBus: EventBus;

    constructor(config: EngineConfig) {
        this.actionRegistry = new Map();
        this.stateStore = config.stateStore;
        this.eventBus = config.eventBus;
        this.registerBuiltinActions();
    }

    registerAction(type: string, handler: ActionHandler): void {
        this.actionRegistry.set(type, handler);
    }

    async execute(workflow: Workflow, input: any): Promise<ExecutionResult> {
        const context = this.createContext(workflow, input);
        
        try {
            await this.stateStore.save(context);
            this.eventBus.emit('workflow:started', { workflow, context });

            // 执行步骤
            for (const step of this.getExecutionOrder(workflow.steps)) {
                if (context.status === 'cancelled') break;
                
                await this.executeStep(step, context);
            }

            context.status = 'completed';
            await this.stateStore.save(context);
            this.eventBus.emit('workflow:completed', { workflow, context });

            return {
                success: true,
                output: this.extractOutput(context),
                executionId: context.executionId
            };
        } catch (error) {
            context.status = 'failed';
            await this.stateStore.save(context);
            
            await this.handleError(workflow.onError, error, context);
            
            return {
                success: false,
                error: error.message,
                executionId: context.executionId
            };
        }
    }

    private async executeStep(
        step: WorkflowStep, 
        context: WorkflowContext
    ): Promise<void> {
        // 检查条件
        if (step.condition && !this.evaluateCondition(step.condition, context)) {
            context.stepResults.set(step.id, { status: 'skipped' });
            return;
        }

        // 检查依赖
        if (step.dependencies) {
            for (const depId of step.dependencies) {
                const depResult = context.stepResults.get(depId);
                if (!depResult || depResult.status !== 'completed') {
                    throw new DependencyNotMetError(step.id, depId);
                }
            }
        }

        context.currentStep = step.id;
        this.eventBus.emit('step:started', { step, context });

        try {
            const handler = this.actionRegistry.get(step.action.type);
            if (!handler) {
                throw new UnknownActionError(step.action.type);
            }

            const params = this.resolveParams(step.action.params, context);
            
            const result = await this.withRetry(
                () => handler(params, context),
                step.retryPolicy
            );

            context.stepResults.set(step.id, {
                status: 'completed',
                output: result
            });

            if (step.action.output) {
                context.variables.set(step.action.output, result);
            }

            this.eventBus.emit('step:completed', { step, context, result });
        } catch (error) {
            context.stepResults.set(step.id, {
                status: 'failed',
                error: error.message
            });
            throw error;
        }
    }

    private getExecutionOrder(steps: WorkflowStep[]): WorkflowStep[] {
        // 拓扑排序确保依赖顺序
        const sorted: WorkflowStep[] = [];
        const visited = new Set<string>();
        const visiting = new Set<string>();

        const visit = (step: WorkflowStep) => {
            if (visited.has(step.id)) return;
            if (visiting.has(step.id)) {
                throw new CircularDependencyError(step.id);
            }

            visiting.add(step.id);

            if (step.dependencies) {
                for (const depId of step.dependencies) {
                    const dep = steps.find(s => s.id === depId);
                    if (dep) visit(dep);
                }
            }

            visiting.delete(step.id);
            visited.add(step.id);
            sorted.push(step);
        };

        for (const step of steps) {
            visit(step);
        }

        return sorted;
    }
}

第二步：内置动作实现

2.1 文件操作动作

// src/actions/file.ts
export class FileActions {
    static read: ActionHandler = async (params, context) => {
        const { path, encoding = 'utf-8' } = params;
        const content = await fs.readFile(path, encoding);
        return content;
    };

    static write: ActionHandler = async (params, context) => {
        const { path, content, encoding = 'utf-8' } = params;
        await fs.writeFile(path, content, encoding);
        return { path, size: content.length };
    };

    static copy: ActionHandler = async (params, context) => {
        const { source, destination } = params;
        await fs.copyFile(source, destination);
        return { source, destination };
    };

    static delete: ActionHandler = async (params, context) => {
        const { path, recursive = false } = params;
        if (recursive) {
            await fs.rm(path, { recursive: true });
        } else {
            await fs.unlink(path);
        }
        return { deleted: path };
    };

    static glob: ActionHandler = async (params, context) => {
        const { pattern, cwd = process.cwd() } = params;
        const files = await glob(pattern, { cwd });
        return files;
    };
}

2.2 Shell 命令动作

// src/actions/shell.ts
export class ShellActions {
    static exec: ActionHandler = async (params, context) => {
        const { 
            command, 
            args = [], 
            cwd, 
            env = {}, 
            timeout = 300000 
        } = params;

        return new Promise((resolve, reject) => {
            const proc = spawn(command, args, {
                cwd,
                env: { ...process.env, ...env },
                stdio: ['ignore', 'pipe', 'pipe']
            });

            let stdout = '';
            let stderr = '';

            proc.stdout.on('data', (data) => {
                stdout += data;
                context.eventBus.emit('shell:stdout', { data: data.toString() });
            });

            proc.stderr.on('data', (data) => {
                stderr += data;
                context.eventBus.emit('shell:stderr', { data: data.toString() });
            });

            const timer = setTimeout(() => {
                proc.kill();
                reject(new TimeoutError(`Command timed out: ${command}`));
            }, timeout);

            proc.on('close', (code) => {
                clearTimeout(timer);
                if (code === 0) {
                    resolve({ stdout, stderr, exitCode: code });
                } else {
                    reject new ShellError(command, code, stdout, stderr);
                }
            });

            proc.on('error', (err) => {
                clearTimeout(timer);
                reject(err);
            });
        });
    };

    static script: ActionHandler = async (params, context) => {
        const { content, interpreter = '/bin/bash' } = params;
        const tmpFile = path.join(os.tmpdir(), `script-${Date.now()}.sh`);
        
        await fs.writeFile(tmpFile, content);
        await fs.chmod(tmpFile, '755');

        try {
            return await ShellActions.exec({
                command: interpreter,
                args: [tmpFile]
            }, context);
        } finally {
            await fs.unlink(tmpFile);
        }
    };
}

2.3 Git 操作动作

// src/actions/git.ts
export class GitActions {
    static clone: ActionHandler = async (params, context) => {
        const { url, directory, branch, depth } = params;
        const args = ['clone'];
        
        if (branch) args.push('-b', branch);
        if (depth) args.push('--depth', depth.toString());
        
        args.push(url, directory);

        return ShellActions.exec({
            command: 'git',
            args
        }, context);
    };

    static checkout: ActionHandler = async (params, context) => {
        const { branch, create = false, cwd } = params;
        const args = ['checkout'];
        
        if (create) args.push('-b');
        args.push(branch);

        return ShellActions.exec({
            command: 'git',
            args,
            cwd
        }, context);
    };

    static commit: ActionHandler = async (params, context) => {
        const { message, cwd, addAll = false } = params;

        if (addAll) {
            await ShellActions.exec({
                command: 'git',
                args: ['add', '-A'],
                cwd
            }, context);
        }

        return ShellActions.exec({
            command: 'git',
            args: ['commit', '-m', message],
            cwd
        }, context);
    };

    static push: ActionHandler = async (params, context) => {
        const { remote = 'origin', branch, cwd, force = false } = params;
        const args = ['push'];
        
        if (force) args.push('-f');
        args.push(remote, branch);

        return ShellActions.exec({
            command: 'git',
            args,
            cwd
        }, context);
    };

    static pr: ActionHandler = async (params, context) => {
        const { title, body, base, head, draft = false } = params;
        
        const args = ['pr', 'create', '--title', title, '--body', body];
        if (base) args.push('--base', base);
        if (head) args.push('--head', head);
        if (draft) args.push('--draft');

        return ShellActions.exec({
            command: 'gh',
            args
        }, context);
    };
}

# workflows/feature.yaml
name: Feature Development
description: 完整的功能开发工作流
version: 1.0.0

triggers:
  - type: command
    command: /workflow feature

variables:
  branch_prefix: feature
  base_branch: main

steps:
  - id: validate
    name: 验证输入参数
    type: task
    action:
      type: validate
      params:
        schema:
          feature_name: { type: string, required: true }
          description: { type: string, required: true }

  - id: create_branch
    name: 创建功能分支
    type: task
    dependencies: [validate]
    action:
      type: git.checkout
      params:
        branch: "${variables.branch_prefix}/${variables.feature_name}"
        create: true

  - id: create_issue
    name: 创建关联 Issue
    type: task
    dependencies: [validate]
    action:
      type: github.createIssue
      params:
        title: "Feature: ${variables.feature_name}"
        body: "${variables.description}"
        labels: [enhancement]
      output: issue

  - id: setup
    name: 设置开发环境
    type: task
    dependencies: [create_branch]
    action:
      type: shell.exec
      params:
        command: npm
        args: [install]

  - id: notify_start
    name: 通知开始开发
    type: task
    dependencies: [create_issue, create_branch]
    action:
      type: slack.send
      params:
        channel: development
        message:
          title: "🚀 开始新功能开发"
          text: "${variables.feature_name}"
          fields:
            - label: "Issue"
              value: "#${steps.create_issue.output.number}"
            - label: "分支"
              value: "${variables.branch_prefix}/${variables.feature_name}"

onError:
  type: notify
  channel: development
  message: "工作流执行失败: ${error.message}"

name: Code Review
description: 自动化代码审查工作流
version: 1.0.0

triggers:
  - type: command
    command: /workflow review

steps:
  - id: run_lint
    name: 运行代码检查
    type: task
    action:
      type: shell.exec
      params:
        command: npm
        args: [run, lint]
    retryPolicy:
      maxAttempts: 1

  - id: run_tests
    name: 运行测试
    type: task
    action:
      type: shell.exec
      params:
        command: npm
        args: [test, --coverage]
    timeout: 300000

  - id: analyze_coverage
    name: 分析覆盖率
    type: task
    dependencies: [run_tests]
    condition:
      type: expression
      value: "${steps.run_tests.output.exitCode} === 0"
    action:
      type: coverage.analyze
      params:
        threshold: 80
        reportPath: coverage/coverage-summary.json

  - id: security_scan
    name: 安全扫描
    type: parallel
    steps:
      - id: npm_audit
        action:
          type: shell.exec
          params:
            command: npm
            args: [audit]
      - id: snyk_scan
        action:
          type: shell.exec
          params:
            command: snyk
            args: [test]

  - id: create_pr
    name: 创建 Pull Request
    type: task
    dependencies: [run_lint, run_tests, analyze_coverage]
    condition:
      type: all
      conditions:
        - type: expression
          value: "${steps.run_lint.output.exitCode} === 0"
        - type: expression
          value: "${steps.run_tests.output.exitCode} === 0"
        - type: expression
          value: "${steps.analyze_coverage.output.passed} === true"
    action:
      type: git.pr
      params:
        title: "${variables.pr_title}"
        body: |
          ## 变更说明
          ${variables.pr_description}
          
          ## 检查结果
          - ✅ Lint 通过
          - ✅ 测试通过
          - ✅ 覆盖率: ${steps.analyze_coverage.output.coverage}%
        base: main
        draft: false

  - id: request_review
    name: 请求代码审查
    type: task
    dependencies: [create_pr]
    action:
      type: github.requestReview
      params:
        pr: "${steps.create_pr.output.number}"
        reviewers: "${variables.reviewers}"

  - id: notify
    name: 发送通知
    type: task
    dependencies: [create_pr]
    action:
      type: slack.send
      params:
        channel: code-review
        message:
          title: "📝 新 PR 待审查"
          text: "${steps.create_pr.output.title}"
          actions:
            - label: "查看 PR"
              url: "${steps.create_pr.output.url}"

第四步：插件集成

4.1 注册工作流命令

// src/plugin.ts
export class WorkflowPlugin implements ClaudeCodePlugin {
    private engine: WorkflowEngine;
    private workflows: Map<string, Workflow>;

    async initialize(context: PluginContext): Promise<void> {
        this.engine = new WorkflowEngine({
            stateStore: new FileStateStore(context.dataDir),
            eventBus: context.eventBus
        });

        this.registerActions();
        await this.loadWorkflows(context.configDir);
        this.registerCommands(context);
    }

    private registerActions(): void {
        // 文件操作
        this.engine.registerAction('file.read', FileActions.read);
        this.engine.registerAction('file.write', FileActions.write);
        this.engine.registerAction('file.copy', FileActions.copy);
        this.engine.registerAction('file.delete', FileActions.delete);
        this.engine.registerAction('file.glob', FileActions.glob);

        // Shell 命令
        this.engine.registerAction('shell.exec', ShellActions.exec);
        this.engine.registerAction('shell.script', ShellActions.script);

        // Git 操作
        this.engine.registerAction('git.clone', GitActions.clone);
        this.engine.registerAction('git.checkout', GitActions.checkout);
        this.engine.registerAction('git.commit', GitActions.commit);
        this.engine.registerAction('git.push', GitActions.push);
        this.engine.registerAction('git.pr', GitActions.pr);
    }

    private registerCommands(context: PluginContext): void {
        context.registerCommand('workflow', {
            description: '管理工作流',
            subCommands: {
                list: {
                    description: '列出所有工作流',
                    execute: async () => {
                        return Array.from(this.workflows.values())
                            .map(w => `- ${w.name}: ${w.description}`)
                            .join('\n');
                    }
                },
                run: {
                    description: '执行工作流',
                    execute: async (args) => {
                        const workflow = this.workflows.get(args.name);
                        if (!workflow) {
                            throw new Error(`工作流不存在: ${args.name}`);
                        }
                        
                        const result = await this.engine.execute(workflow, args);
                        return result.success 
                            ? `工作流执行成功: ${result.executionId}`
                            : `工作流执行失败: ${result.error}`;
                    }
                },
                status: {
                    description: '查看执行状态',
                    execute: async (args) => {
                        const status = await this.engine.getStatus(args.executionId);
                        return JSON.stringify(status, null, 2);
                    }
                },
                cancel: {
                    description: '取消执行',
                    execute: async (args) => {
                        await this.engine.cancel(args.executionId);
                        return '工作流已取消';
                    }
                }
            }
        });
    }
}

最佳实践

工作流设计原则

  • 单一职责：每个步骤只做一件事
  • 幂等性：工作流可以安全地重复执行
  • 可观测性：记录详细的执行日志
  • 可恢复性：支持从失败点恢复执行

错误处理策略

  • 定义明确的错误处理流程
  • 实现自动重试机制
  • 提供人工干预入口
  • 发送失败通知

性能优化

  • 使用并行步骤加速执行
  • 实现步骤缓存避免重复计算
  • 合理设置超时时间
  • 优化依赖关系减少等待
