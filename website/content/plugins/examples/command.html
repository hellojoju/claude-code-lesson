插件概述

本案例实现一个代码片段管理插件，支持保存、搜索和插入代码片段。

功能特性

  • 保存代码片段
  • 搜索代码片段
  • 插入代码片段
  • 管理代码片段

实现代码

// src/index.ts
import { Plugin, PluginContext } from '@anthropic/claude-code-api';
import * as fs from 'fs';
import * as path from 'path';

interface Snippet {
  id: string;
  name: string;
  language: string;
  code: string;
  tags: string[];
  createdAt: Date;
}

class SnippetManager {
  private snippets: Map = new Map();
  private storagePath: string;
  
  constructor(storagePath: string) {
    this.storagePath = storagePath;
    this.load();
  }
  
  private load() {
    const file = path.join(this.storagePath, 'snippets.json');
    if (fs.existsSync(file)) {
      const data = JSON.parse(fs.readFileSync(file, 'utf8'));
      data.forEach((s: Snippet) => this.snippets.set(s.id, s));
    }
  }
  
  private save() {
    const file = path.join(this.storagePath, 'snippets.json');
    fs.writeFileSync(file, JSON.stringify([...this.snippets.values()]));
  }
  
  add(snippet: Omit): Snippet {
    const id = Date.now().toString();
    const newSnippet = {
      ...snippet,
      id,
      createdAt: new Date()
    };
    this.snippets.set(id, newSnippet);
    this.save();
    return newSnippet;
  }
  
  search(query: string): Snippet[] {
    const lower = query.toLowerCase();
    return [...this.snippets.values()].filter(s =>
      s.name.toLowerCase().includes(lower) ||
      s.tags.some(t => t.toLowerCase().includes(lower))
    );
  }
  
  get(id: string): Snippet | undefined {
    return this.snippets.get(id);
  }
  
  list(): Snippet[] {
    return [...this.snippets.values()];
  }
  
  delete(id: string): boolean {
    const result = this.snippets.delete(id);
    if (result) this.save();
    return result;
  }
}

let manager: SnippetManager;

export const plugin: Plugin = {
  async activate(context: PluginContext) {
    const storagePath = context.getConfiguration('snippets').get('storagePath');
    
    // 确保存储目录存在
    if (!fs.existsSync(storagePath)) {
      fs.mkdirSync(storagePath, { recursive: true });
    }
    
    manager = new SnippetManager(storagePath);
    
    // 注册命令
    context.registerCommand('snippets.save', async () => {
      const name = await context.showInputBox({
        prompt: '片段名称',
        placeHolder: 'my-snippet'
      });
      
      if (!name) return;
      
      const code = await context.showInputBox({
        prompt: '代码内容',
        placeHolder: 'console.log("hello");'
      });
      
      if (!code) return;
      
      const language = await context.showQuickPick(
        ['JavaScript', 'TypeScript', 'Python', 'Go', 'Rust'],
        { placeHolder: '选择语言' }
      );
      
      const tagsInput = await context.showInputBox({
        prompt: '标签（逗号分隔）',
        placeHolder: 'utility, helper'
      });
      
      const tags = tagsInput ? tagsInput.split(',').map(t => t.trim()) : [];
      
      const snippet = manager.add({ name, code, language, tags });
      
      context.showInformationMessage(`片段 "${snippet.name}" 已保存`);
    });
    
    context.registerCommand('snippets.search', async () => {
      const query = await context.showInputBox({
        prompt: '搜索片段',
        placeHolder: '输入关键词'
      });
      
      if (!query) return;
      
      const results = manager.search(query);
      
      if (results.length === 0) {
        context.showInformationMessage('没有找到匹配的片段');
        return;
      }
      
      const selected = await context.showQuickPick(
        results.map(s => ({
          label: s.name,
          description: s.language,
          detail: s.code.substring(0, 50) + '...',
          snippet: s
        })),
        { placeHolder: '选择片段' }
      );
      
      if (selected) {
        // 插入选中的片段
        context.showInformationMessage(`已选择: ${selected.snippet.name}`);
      }
    });
    
    context.registerCommand('snippets.list', async () => {
      const snippets = manager.list();
      
      if (snippets.length === 0) {
        context.showInformationMessage('没有保存的片段');
        return;
      }
      
      const selected = await context.showQuickPick(
        snippets.map(s => ({
          label: s.name,
          description: s.language,
          detail: s.tags.join(', ')
        })),
        { placeHolder: '所有片段' }
      );
    });
  },
  
  async deactivate() {
    // 清理资源
  }
};

export default plugin;

使用示例

# 保存代码片段
> /snippets.save

# 搜索片段
> /snippets.search

# 列出所有片段
> /snippets.list
