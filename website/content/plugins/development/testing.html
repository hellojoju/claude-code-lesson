单元测试

测试命令

// tests/commands/hello.test.ts
import { createMockContext } from '@anthropic/claude-code-test';
import { plugin } from '../../src/index';

describe('hello command', () => {
  let mockContext: any;
  
  beforeEach(() => {
    mockContext = createMockContext();
  });
  
  it('should show greeting message', async () => {
    // 模拟用户输入
    mockContext.showInputBox.mockResolvedValue('张三');
    
    // 激活插件
    await plugin.activate(mockContext);
    
    // 获取注册的命令
    const helloCommand = mockContext.getCommand('myPlugin.hello');
    
    // 执行命令
    await helloCommand.handler();
    
    // 验证消息
    expect(mockContext.showInformationMessage).toHaveBeenCalledWith(
      '你好，张三！欢迎使用我的插件。'
    );
  });
  
  it('should handle empty input', async () => {
    mockContext.showInputBox.mockResolvedValue('');
    
    await plugin.activate(mockContext);
    const helloCommand = mockContext.getCommand('myPlugin.hello');
    
    await helloCommand.handler();
    
    expect(mockContext.showWarningMessage).toHaveBeenCalled();
  });
});

集成测试

测试插件激活

// tests/integration/activation.test.ts
import { PluginTestRunner } from '@anthropic/claude-code-test';
import path from 'path';

describe('Plugin Activation', () => {
  let runner: PluginTestRunner;
  
  beforeAll(async () => {
    runner = new PluginTestRunner({
      extensionPath: path.resolve(__dirname, '..')
    });
    
    await runner.start();
  });
  
  afterAll(async () => {
    await runner.stop();
  });
  
  it('should activate successfully', async () => {
    const result = await runner.activate();
    expect(result.success).toBe(true);
  });
  
  it('should register all commands', async () => {
    const commands = await runner.getCommands();
    expect(commands).toContain('myPlugin.hello');
    expect(commands).toContain('myPlugin.stats');
  });
  
  it('should execute command', async () => {
    const result = await runner.executeCommand('myPlugin.hello');
    expect(result.success).toBe(true);
  });
});

    "test:coverage": "jest --coverage"
  }

}

CI/CD 集成

GitHub Actions

# .github/workflows/test.yml
name: Test

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          
      - name: Install dependencies
        run: npm ci
        
      - name: Run tests
        run: npm test
        
      - name: Upload coverage
        uses: codecov/codecov-action@v3
