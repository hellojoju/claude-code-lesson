基础插件开发

来源说明：本文内容基于 Claude Code 官方文档和社区最佳实践整理。

可靠性评级：A 官方文档 + 社区验证

快速开始

创建插件项目

# 创建项目目录
mkdir my-first-plugin && cd my-first-plugin

# 初始化 npm
npm init -y

# 安装开发依赖
npm install --save-dev typescript @anthropic/claude-code-api

# 创建源码目录
mkdir -p src

创建入口文件

// src/index.ts
import { Plugin, PluginContext } from '@anthropic/claude-code-api';

export const plugin: Plugin = {
  async activate(context: PluginContext) {
    console.log('我的第一个插件已激活！');
  },
  
  async deactivate() {
    console.log('插件已停用');
  }
};

export default plugin;

配置 package.json

{
  "name": "claude-code-plugin-my-first",
  "version": "1.0.0",
  "description": "我的第一个 Claude Code 插件",
  "main": "dist/index.js",
  "scripts": {
    "build": "tsc",
    "watch": "tsc -w"
  },
  "claudeCode": {
    "name": "my-first-plugin",
    "displayName": "我的第一个插件",
    "description": "学习插件开发的示例",
    "version": "1.0.0",
    "engines": {
      "claudeCode": ">=1.0.0"
    },
    "categories": ["examples"],
    "activationEvents": ["*"],
    "contributes": {
      "commands": [
        {
          "command": "myFirstPlugin.hello",
          "title": "说你好"
        }
      ]
    }
  }
}

注册命令

命令处理器

// src/index.ts
import { Plugin, PluginContext } from '@anthropic/claude-code-api';

export const plugin: Plugin = {
  async activate(context: PluginContext) {
    // 注册命令
    context.registerCommand('myFirstPlugin.hello', async () => {
      const name = await context.showInputBox({
        prompt: '请输入你的名字',
        placeHolder: '张三'
      });
      
      context.showInformationMessage(`你好，${name}！欢迎使用我的插件。`);
    });
  },
  
  async deactivate() {
    // 清理资源
  }
};

使用命令

# 在 Claude Code 中使用
> /myFirstPlugin.hello

注册技能

技能配置

// package.json
{
  "contributes": {
    "skills": [
      {
        "id": "myFirstPlugin.greet",
        "name": "问候",
        "description": "生成问候语",
        "triggers": ["问候", "greet", "hello"]
      }
    ]
  }
}

技能实现

// src/skills/greet.ts
export async function execute(context: SkillContext) {
  const { name } = context.params;
  
  return {
    message: `你好，${name}！很高兴认识你。`
  };
}

状态管理

使用状态存储

// src/index.ts
export const plugin: Plugin = {
  async activate(context: PluginContext) {
    // 读取状态
    const count = context.globalState.get('count', 0);
    
    // 更新状态
    context.globalState.set('count', count + 1);
    
    // 工作区状态（仅当前项目）
    const projectConfig = context.workspaceState.get('config', {});
    
    // 注册命令使用状态
    context.registerCommand('myFirstPlugin.stats', () => {
      context.showInformationMessage(`插件已激活 ${count} 次`);
    });
  }
};

编译和测试

TypeScript 配置

// tsconfig.json
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true
  },
  "include": ["src/**/*"]
}

编译和安装

# 编译
npm run build

# 本地安装测试
> /plugin install /path/to/my-first-plugin

# 测试命令
> /myFirstPlugin.hello
