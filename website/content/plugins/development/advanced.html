异步操作

异步命令

context.registerCommand('myPlugin.fetchData', async () => {
  // 显示进度
  await context.withProgress({
    title: '获取数据...',
    location: 'notification'
  }, async (progress) => {
    progress.report({ message: '连接服务器...' });
    const response = await fetch('https://api.example.com/data');
    
    progress.report({ message: '解析数据...' });
    const data = await response.json();
    
    return data;
  });
});

取消支持

context.registerCommand('myPlugin.longTask', async (token: CancellationToken) => {
  for (let i = 0; i < 100; i++) {
    // 检查是否被取消
    if (token.isCancellationRequested) {
      return;
    }
    
    await processItem(i);
  }
});

语言服务器

集成语言服务器

import { LanguageClient } from '@anthropic/claude-code-languageclient';

let client: LanguageClient;

export async function activate(context: PluginContext) {
  const serverPath = path.join(context.extensionPath, 'server', 'index.js');
  
  client = new LanguageClient(
    'myLanguageServer',
    'My Language Server',
    {
      module: serverPath,
      transport: 'ipc'
    },
    {
      documentSelector: [{ language: 'mylang' }],
      synchronize: {
        configurationSection: 'myLang'
      }
    }
  );
  
  await client.start();
}

export async function deactivate() {
  await client.stop();
}
