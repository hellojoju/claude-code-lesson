React 组件开发案例

可靠性: A 来源: Claude Code 实战经验

场景描述

开发一个可复用的数据表格组件，支持排序、筛选、分页和行选择功能。这是一个常见但功能丰富的组件，适合展示 Claude Code 在组件开发中的能力。

问题分析

需求拆解

  • 基础表格渲染
  • 列排序（升序/降序）
  • 列筛选（文本搜索/选择过滤）
  • 分页控制
  • 行选择（单选/多选）
  • 自定义单元格渲染
  • 响应式布局

技术选型

  • React 18 + TypeScript
  • Tailwind CSS 样式
  • 无外部表格库依赖

// types.ts
export interface Column<T> {
    key: keyof T;
    title: string;
    width?: number;
    sortable?: boolean;
    filterable?: boolean;
    render?: (value: T[keyof T], record: T) => React.ReactNode;
}

export interface TableProps<T> {
    columns: Column<T>[];
    data: T[];
    rowKey: keyof T;
    pagination?: PaginationConfig;
    selection?: SelectionConfig<T>;
    loading?: boolean;
    onRowClick?: (record: T) => void;
}

export interface PaginationConfig {
    current: number;
    pageSize: number;
    total: number;
    onChange: (page: number, pageSize: number) => void;
}

export interface SelectionConfig<T> {
    type: 'single' | 'multiple';
    selectedKeys: (T[keyof T])[];
    onChange: (selectedKeys: (T[keyof T])[], selectedRows: T[]) => void;
}

export interface SortState {
    key: string;
    order: 'asc' | 'desc';
}

export interface FilterState {
    [key: string]: string | string[];
}

// DataTable.tsx
import { useState, useMemo } from 'react';
import { TableProps, SortState, FilterState } from './types';

export function DataTable<T extends Record<string, any>>({
    columns,
    data,
    rowKey,
    pagination,
    selection,
    loading = false,
    onRowClick
}: TableProps<T>) {
    const [sort, setSort] = useState<SortState | null>(null);
    const [filters, setFilters] = useState<FilterState>({});

    // 处理排序
    const handleSort = (key: string) => {
        setSort(prev => {
            if (!prev || prev.key !== key) {
                return { key, order: 'asc' };
            }
            if (prev.order === 'asc') {
                return { key, order: 'desc' };
            }
            return null;
        });
    };

    // 处理筛选
    const handleFilter = (key: string, value: string | string[]) => {
        setFilters(prev => ({
            ...prev,
            [key]: value
        }));
    };

    // 计算显示数据
    const displayData = useMemo(() => {
        let result = [...data];

        // 应用筛选
        Object.entries(filters).forEach(([key, value]) => {
            if (value) {
                result = result.filter(item => {
                    const itemValue = String(item[key]).toLowerCase();
                    const filterValue = Array.isArray(value) 
                        ? value 
                        : [value];
                    return filterValue.some(v => 
                        itemValue.includes(v.toLowerCase())
                    );
                });
            }
        });

        // 应用排序
        if (sort) {
            result.sort((a, b) => {
                const aVal = a[sort.key];
                const bVal = b[sort.key];
                const compare = aVal < bVal ? -1 : aVal > bVal ? 1 : 0;
                return sort.order === 'desc' ? -compare : compare;
            });
        }

        // 应用分页
        if (pagination) {
            const start = (pagination.current - 1) * pagination.pageSize;
            result = result.slice(start, start + pagination.pageSize);
        }

        return result;
    }, [data, sort, filters, pagination]);

    // 渲染表头
    const renderHeader = () => (
        <thead className="bg-gray-50">
            <tr>
                {selection && (
                    <th className="px-4 py-3 w-12">
                        {selection.type === 'multiple' && (
                            <input
                                type="checkbox"
                                onChange={handleSelectAll}
                                checked={isAllSelected}
                            />
                        )}
                    </th>
                )}
                {columns.map(col => (
                    <th
                        key={String(col.key)}
                        className="px-4 py-3 text-left"
                        style={{ width: col.width }}
                    >
                        <div className="flex items-center gap-2">
                            <span>{col.title}</span>
                            {col.sortable && (
                                <button
                                    onClick={() => handleSort(String(col.key))}
                                    className="text-gray-400 hover:text-gray-600"
                                >
                                    {getSortIcon(String(col.key))}
                                </button>
                            )}
                        </div>
                    </th>
                ))}
            </tr>
        </thead>
    );

    // 渲染表格体
    const renderBody = () => {
        if (loading) {
            return (
                <tbody>
                    <tr>
                        <td
                            colSpan={columns.length + (selection ? 1 : 0)}
                            className="px-4 py-8 text-center"
                        >
                            <span className="animate-pulse">加载中...</span>
                        </td>
                    </tr>
                </tbody>
            );
        }

        return (
            <tbody className="divide-y">
                {displayData.map(record => (
                    <tr
                        key={String(record[rowKey])}
                        className="hover:bg-gray-50 cursor-pointer"
                        onClick={() => onRowClick?.(record)}
                    >
                        {selection && (
                            <td className="px-4 py-3">
                                <input
                                    type={selection.type === 'multiple' 
                                        ? 'checkbox' 
                                        : 'radio'}
                                    checked={isSelected(record[rowKey])}
                                    onChange={() => handleSelect(record)}
                                    onClick={e => e.stopPropagation()}
                                />
                            </td>
                        )}
                        {columns.map(col => (
                            <td key={String(col.key)} className="px-4 py-3">
                                {col.render
                                    ? col.render(record[col.key], record)
                                    : String(record[col.key])}
                            </td>
                        ))}
                    </tr>
                ))}
            </tbody>
        );
    };

    return (
        <div className="overflow-x-auto">
            <table className="min-w-full divide-y">
                {renderHeader()}
                {renderBody()}
            </table>
            {pagination && <Pagination {...pagination} />}
        </div>
    );
}

第三步：添加分页组件

// Pagination.tsx
interface PaginationProps {
    current: number;
    pageSize: number;
    total: number;
    onChange: (page: number, pageSize: number) => void;
}

function Pagination({ current, pageSize, total, onChange }: PaginationProps) {
    const totalPages = Math.ceil(total / pageSize);

    return (
        <div className="flex items-center justify-between px-4 py-3">
            <div className="text-sm text-gray-500">
                共 {total} 条记录
            </div>
            <div className="flex items-center gap-2">
                <button
                    onClick={() => onChange(current - 1, pageSize)}
                    disabled={current === 1}
                    className="px-3 py-1 rounded disabled:opacity-50"
                >
                    上一页
                </button>
                {renderPageNumbers()}
                <button
                    onClick={() => onChange(current + 1, pageSize)}
                    disabled={current === totalPages}
                    className="px-3 py-1 rounded disabled:opacity-50"
                >
                    下一页
                </button>
            </div>
        </div>
    );
}

import { DataTable } from './components/DataTable';

interface User {
    id: string;
    name: string;
    email: string;
    role: string;
    createdAt: string;
}

function UserList() {
    const [users, setUsers] = useState<User[]>([]);
    const [pagination, setPagination] = useState({
        current: 1,
        pageSize: 10,
        total: 0
    });
    const [selectedKeys, setSelectedKeys] = useState<string[]>([]);

    const columns: Column<User>[] = [
        {
            key: 'name',
            title: '姓名',
            sortable: true,
            filterable: true
        },
        {
            key: 'email',
            title: '邮箱',
            sortable: true
        },
        {
            key: 'role',
            title: '角色',
            render: (role) => (
                <span className={`px-2 py-1 rounded ${
                    role === 'admin' ? 'bg-blue-100 text-blue-800' : 'bg-gray-100'
                }`}>
                    {role}
                </span>
            )
        },
        {
            key: 'createdAt',
            title: '创建时间',
            sortable: true,
            render: (date) => new Date(date).toLocaleDateString()
        }
    ];

    return (
        <DataTable
            columns={columns}
            data={users}
            rowKey="id"
            pagination={{
                ...pagination,
                onChange: (page, pageSize) => {
                    setPagination(prev => ({ ...prev, current: page, pageSize }));
                    fetchUsers(page, pageSize);
                }
            }}
            selection={{
                type: 'multiple',
                selectedKeys,
                onChange: (keys, rows) => setSelectedKeys(keys as string[])
            }}
            onRowClick={(user) => console.log('Clicked:', user)}
        />
    );
}

最佳实践

组件设计原则

  • 单一职责：每个组件只负责一件事
  • 可配置性：通过 props 控制行为
  • 可扩展性：支持自定义渲染
  • 类型安全：完整的 TypeScript 类型

性能优化

  • 使用 useMemo 缓存计算结果
  • 使用 useCallback 缓存回调函数
  • 虚拟滚动处理大数据量
  • 按需加载表格数据
