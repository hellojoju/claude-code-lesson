RESTful API 开发案例

可靠性: A 来源: Claude Code 实战经验

场景描述

为一个博客系统设计和实现 RESTful API，包括文章、评论、用户等资源的 CRUD 操作。

API 设计

资源设计

文章资源 /articles
- GET    /articles           # 获取文章列表
- GET    /articles/:id       # 获取单篇文章
- POST   /articles           # 创建文章
- PUT    /articles/:id       # 更新文章
- DELETE /articles/:id       # 删除文章

评论资源 /articles/:id/comments
- GET    /articles/:id/comments    # 获取文章评论
- POST   /articles/:id/comments    # 添加评论

用户资源 /users
- GET    /users/:id         # 获取用户信息
- PUT    /users/:id         # 更新用户信息

响应格式

// 成功响应
{
  "code": 0,
  "data": { ... },
  "message": "success"
}

// 错误响应
{
  "code": 1001,
  "data": null,
  "message": "参数无效"
}

第二步：数据库模型

// prisma/schema.prisma
model User {
    id        String   @id @default(cuid())
    email     String   @unique
    name      String
    password  String
    articles  Article[]
    comments  Comment[]
    createdAt DateTime @default(now())
    updatedAt DateTime @updatedAt
}

model Article {
    id          String    @id @default(cuid())
    title       String
    content     String
    summary     String?
    authorId    String
    author      User      @relation(fields: [authorId], references: [id])
    comments    Comment[]
    status      Status    @default(DRAFT)
    publishedAt DateTime?
    createdAt   DateTime  @default(now())
    updatedAt   DateTime  @updatedAt

    @@index([authorId])
    @@index([status])
}

model Comment {
    id        String   @id @default(cuid())
    content   String
    articleId String
    article   Article  @relation(fields: [articleId], references: [id])
    authorId  String
    author    User     @relation(fields: [authorId], references: [id])
    createdAt DateTime @default(now())
    updatedAt DateTime @updatedAt

    @@index([articleId])
}

enum Status {
    DRAFT
    PUBLISHED
    ARCHIVED
}

第三步：路由实现

// routes/articles.ts
import { Router } from 'express';
import { body, query, param } from 'express-validator';
import { validate } from '../middlewares/validate';
import { auth } from '../middlewares/auth';
import * as articleController from '../controllers/articles';

const router = Router();

// 获取文章列表
router.get('/',
    query('page').optional().isInt({ min: 1 }),
    query('limit').optional().isInt({ min: 1, max: 100 }),
    query('status').optional().isIn(['DRAFT', 'PUBLISHED', 'ARCHIVED']),
    validate,
    articleController.list
);

// 获取单篇文章
router.get('/:id',
    param('id').isString().notEmpty(),
    validate,
    articleController.get
);

// 创建文章
router.post('/',
    auth,
    body('title').isString().trim().isLength({ min: 1, max: 200 }),
    body('content').isString().trim().isLength({ min: 1 }),
    body('summary').optional().isString().trim().isLength({ max: 500 }),
    validate,
    articleController.create
);

// 更新文章
router.put('/:id',
    auth,
    param('id').isString().notEmpty(),
    body('title').optional().isString().trim().isLength({ min: 1, max: 200 }),
    body('content').optional().isString().trim().isLength({ min: 1 }),
    validate,
    articleController.update
);

// 删除文章
router.delete('/:id',
    auth,
    param('id').isString().notEmpty(),
    validate,
    articleController.remove
);

export default router;

第四步：控制器实现

// controllers/articles.ts
import { Request, Response, NextFunction } from 'express';
import * as articleService from '../services/articles';

export async function list(req: Request, res: Response, next: NextFunction) {
    try {
        const { page = 1, limit = 10, status } = req.query;

        const result = await articleService.findMany({
            page: Number(page),
            limit: Number(limit),
            status: status as string
        });

        res.json({
            code: 0,
            data: result,
            message: 'success'
        });
    } catch (error) {
        next(error);
    }
}

export async function get(req: Request, res: Response, next: NextFunction) {
    try {
        const { id } = req.params;

        const article = await articleService.findById(id);

        if (!article) {
            return res.status(404).json({
                code: 1004,
                data: null,
                message: '文章不存在'
            });
        }

        res.json({
            code: 0,
            data: article,
            message: 'success'
        });
    } catch (error) {
        next(error);
    }
}

export async function create(req: Request, res: Response, next: NextFunction) {
    try {
        const userId = req.user!.id;
        const { title, content, summary } = req.body;

        const article = await articleService.create({
            title,
            content,
            summary,
            authorId: userId
        });

        res.status(201).json({
            code: 0,
            data: article,
            message: 'success'
        });
    } catch (error) {
        next(error);
    }
}

最佳实践

API 设计原则

  • 使用名词复数表示资源
  • 使用 HTTP 方法表示操作
  • 版本控制 API
  • 统一响应格式
  • 合理使用 HTTP 状态码

安全考虑

  • 验证所有输入参数
  • 实现认证和授权
  • 防止 SQL 注入
  • 实现速率限制
