实时应用开发案例

可靠性: A 来源: Claude Code 实战经验

场景描述

开发一个实时聊天应用，使用 WebSocket 实现消息的实时推送和接收。

WebSocket 服务端

// server.ts
import { Server } from 'socket.io';
import { createServer } from 'http';

const httpServer = createServer();
const io = new Server(httpServer, {
    cors: { origin: '*' }
});

interface Message {
    id: string;
    roomId: string;
    userId: string;
    content: string;
    timestamp: number;
}

const rooms = new Map<string, Set<string>>();

io.on('connection', (socket) => {
    console.log('用户连接:', socket.id);

    // 加入房间
    socket.on('join', (roomId: string, userId: string) => {
        socket.join(roomId);
        
        if (!rooms.has(roomId)) {
            rooms.set(roomId, new Set());
        }
        rooms.get(roomId)!.add(userId);

        // 通知其他用户
        socket.to(roomId).emit('user:joined', { userId });
        
        // 发送在线用户列表
        io.to(roomId).emit('users', Array.from(rooms.get(roomId)!));
    });

    // 接收消息
    socket.on('message', (data: Omit<Message, 'id' | 'timestamp'>) => {
        const message: Message = {
            ...data,
            id: generateId(),
            timestamp: Date.now()
        };

        // 广播到房间
        io.to(data.roomId).emit('message', message);
    });

    // 离开房间
    socket.on('leave', (roomId: string, userId: string) => {
        socket.leave(roomId);
        rooms.get(roomId)?.delete(userId);
        socket.to(roomId).emit('user:left', { userId });
    });

    // 断开连接
    socket.on('disconnect', () => {
        console.log('用户断开:', socket.id);
    });
});

httpServer.listen(3001, () => {
    console.log('WebSocket 服务器运行在 3001 端口');
});

前端实现

// hooks/useChat.ts
import { useEffect, useState, useCallback } from 'react';
import { io, Socket } from 'socket.io-client';

interface Message {
    id: string;
    roomId: string;
    userId: string;
    content: string;
    timestamp: number;
}

export function useChat(roomId: string, userId: string) {
    const [socket, setSocket] = useState<Socket | null>(null);
    const [messages, setMessages] = useState<Message[]>([]);
    const [users, setUsers] = useState<string[]>([]);

    useEffect(() => {
        const newSocket = io('http://localhost:3001');
        setSocket(newSocket);

        newSocket.emit('join', roomId, userId);

        newSocket.on('message', (message: Message) => {
            setMessages(prev => [...prev, message]);
        });

        newSocket.on('users', (userList: string[]) => {
            setUsers(userList);
        });

        return () => {
            newSocket.emit('leave', roomId, userId);
            newSocket.close();
        };
    }, [roomId, userId]);

    const sendMessage = useCallback((content: string) => {
        if (socket) {
            socket.emit('message', { roomId, userId, content });
        }
    }, [socket, roomId, userId]);

    return { messages, users, sendMessage };
}
