创建高级技能

来源说明：本文内容基于 Claude Code 官方文档和社区最佳实践整理。

可靠性评级：A 官方文档 + 社区验证

多步骤技能

定义步骤

{
  "name": "full-review",
  "steps": [
    {
      "name": "analyze",
      "skill": "analyze",
      "inputs": {
        "file": "${inputs.file}"
      }
    },
    {
      "name": "review",
      "skill": "code-review",
      "inputs": {
        "file": "${inputs.file}",
        "focus": "${inputs.focus}"
      }
    },
    {
      "name": "report",
      "prompt": "综合分析和审查结果，生成最终报告"
    }
  ]
}

步骤间数据传递

{
  "steps": [
    {
      "name": "step1",
      "outputs": ["result1"]
    },
    {
      "name": "step2",
      "inputs": {
        "data": "${steps.step1.result1}"
      }
    }
  ]
}

使用模板

模板文件

// templates/component.tsx
import React from 'react';

interface {{ComponentName}}Props {
  {{#each props}}
  {{name}}: {{type}};
  {{/each}}
}

export const {{ComponentName}}: React.FC<{{ComponentName}}Props> = (props) => {
  return (
    <div className="{{className}}">
      {props.children}
    </div>
  );
};

在技能中使用模板

{
  "name": "create-component",
  "inputs": {
    "name": {
      "type": "string",
      "required": true
    },
    "props": {
      "type": "array",
      "default": []
    }
  },
  "template": "templates/component.tsx",
  "output": {
    "type": "file",
    "path": "src/components/{{name}}.tsx"
  }
}

上下文访问

可用上下文

{
  "context": {
    "project": {
      "name": "项目名称",
      "path": "项目路径",
      "tech_stack": ["技术栈"]
    },
    "file": {
      "path": "当前文件路径",
      "content": "文件内容",
      "language": "语言类型"
    },
    "selection": {
      "text": "选中文本",
      "range": "选中范围"
    },
    "git": {
      "branch": "当前分支",
      "status": "Git 状态"
    }
  }
}

在提示词中使用上下文

# prompt.md

项目信息：
- 名称：{{context.project.name}}
- 技术栈：{{context.project.tech_stack}}

当前文件：{{context.file.path}}

请根据项目的技术栈，优化以下代码：
{{selection}}

条件逻辑

条件执行

{
  "steps": [
    {
      "name": "typescript-check",
      "condition": {
        "context.file.language": "typescript"
      },
      "skill": "ts-analyze"
    },
    {
      "name": "javascript-check",
      "condition": {
        "context.file.language": "javascript"
      },
      "skill": "js-analyze"
    }
  ]
}

条件提示词

# prompt.md

{{#if (eq context.file.language "typescript")}}
请使用 TypeScript 最佳实践审查代码。
{{else}}
请使用 JavaScript 最佳实践审查代码。
{{/if}}

技能钩子

生命周期钩子

{
  "hooks": {
    "beforeRun": [
      {
        "action": "validate",
        "params": {
          "file": "${inputs.file}"
        }
      }
    ],
    "afterRun": [
      {
        "action": "save-report",
        "params": {
          "output": "${outputs.report}"
        }
      }
    ],
    "onError": [
      {
        "action": "log-error",
        "params": {
          "error": "${error}"
        }
      }
    ]
  }
}

技能依赖

声明依赖

{
  "name": "full-review",
  "dependencies": {
    "skills": ["analyze", "code-review"],
    "packages": ["prettier", "eslint"]
  }
}

依赖检查

# 安装技能时自动检查依赖
> /skill install full-review

# 检查依赖是否满足
> /skill check-deps full-review

高级示例

完整的项目初始化技能

{
  "name": "init-project",
  "version": "1.0.0",
  "description": "初始化新项目",
  
  "inputs": {
    "name": {
      "type": "string",
      "required": true,
      "description": "项目名称"
    },
    "type": {
      "type": "string",
      "options": ["react", "vue", "node", "next"],
      "default": "react"
    },
    "features": {
      "type": "array",
      "items": "string",
      "options": ["typescript", "eslint", "prettier", "testing"],
      "default": ["typescript", "eslint"]
    }
  },
  
  "steps": [
    {
      "name": "create-structure",
      "action": "create-directories",
      "directories": ["src", "tests", "docs"]
    },
    {
      "name": "generate-config",
      "template": "templates/{{type}}-config.json",
      "output": "package.json"
    },
    {
      "name": "setup-typescript",
      "condition": {
        "inputs.features": { "contains": "typescript" }
      },
      "template": "templates/tsconfig.json",
      "output": "tsconfig.json"
    },
    {
      "name": "setup-eslint",
      "condition": {
        "inputs.features": { "contains": "eslint" }
      },
      "template": "templates/.eslintrc.json",
      "output": ".eslintrc.json"
    },
    {
      "name": "create-readme",
      "template": "templates/README.md",
      "output": "README.md"
    }
  ],
  
  "hooks": {
    "afterRun": [
      {
        "action": "run-command",
        "command": "npm install"
      },
      {
        "action": "git-init"
      }
    ]
  }
}
