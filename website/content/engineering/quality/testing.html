测试模板

// 测试文件模板
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { functionToTest } from './module';

describe('functionToTest', () => {
    describe('正常情况', () => {
        it('应该返回正确结果', () => {
            const result = functionToTest('input');
            expect(result).toBe('expected');
        });
    });

    describe('边界情况', () => {
        it('应该处理空输入', () => {
            expect(() => functionToTest('')).toThrow();
        });

        it('应该处理 null', () => {
            expect(() => functionToTest(null)).toThrow();
        });
    });

    describe('错误情况', () => {
        it('应该抛出有意义的错误', () => {
            expect(() => functionToTest('invalid'))
                .toThrow('Invalid input');
        });
    });
});

集成测试

API 集成测试

// tests/integration/api.test.ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { setupServer } from 'msw/node';
import { rest } from 'msw';

const server = setupServer(
    rest.get('/api/users', (req, res, ctx) => {
        return res(ctx.json({ users: [] }));
    })
);

beforeAll(() => server.listen());
afterAll(() => server.close());

describe('User API', () => {
    it('应该返回用户列表', async () => {
        const response = await fetch('/api/users');
        const data = await response.json();
        
        expect(response.status).toBe(200);
        expect(data.users).toBeDefined();
    });
});

数据库集成测试

// tests/integration/database.test.ts
import { describe, it, expect, beforeEach } from 'vitest';
import { db } from '../src/db';
import { users } from '../src/db/schema';

describe('Database Operations', () => {
    beforeEach(async () => {
        await db.delete(users);
    });

    it('应该创建用户', async () => {
        const user = await db.insert(users).values({
            name: 'Test User',
            email: 'test@example.com'
        }).returning();

        expect(user[0].id).toBeDefined();
        expect(user[0].name).toBe('Test User');
    });
});

E2E 测试

Playwright 测试示例

// e2e/login.spec.ts
import { test, expect } from '@playwright/test';

test.describe('登录流程', () => {
    test('用户可以登录', async ({ page }) => {
        await page.goto('/login');
        
        await page.fill('input[name="email"]', 'user@example.com');
        await page.fill('input[name="password"]', 'password123');
        await page.click('button[type="submit"]');
        
        await expect(page).toHaveURL('/dashboard');
        await expect(page.locator('.user-name')).toBeVisible();
    });

    test('登录失败显示错误信息', async ({ page }) => {
        await page.goto('/login');
        
        await page.fill('input[name="email"]', 'wrong@example.com');
        await page.fill('input[name="password"]', 'wrongpassword');
        await page.click('button[type="submit"]');
        
        await expect(page.locator('.error-message')).toBeVisible();
    });
});

测试最佳实践

测试原则

  • FAST：测试应该快速执行
  • Independent：测试之间相互独立
  • Repeatable：测试结果可重复
  • Self-Validating：测试自动验证结果
  • Timely：及时编写测试

命名规范

// 好的测试命名
it('应该在用户未登录时重定向到登录页', () => {});
it('应该在输入无效邮箱时显示错误提示', () => {});

// 不好的测试命名
it('test1', () => {});
it('works', () => {});

断言技巧

// 使用具体的断言
expect(result).toBe(42);           // 精确匹配
expect(result).toEqual({ a: 1 });  // 深度匹配
expect(result).toContain('text');  // 包含
expect(result).toThrow();          // 抛出异常
expect(result).resolves.toBe(1);   // Promise 成功
expect(result).rejects.toThrow();  // Promise 失败

测试自动化

CI 集成

# .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
    test:
        runs-on: ubuntu-latest
        steps:
            - uses: actions/checkout@v4
            - uses: actions/setup-node@v4
              with:
                  node-version: '20'
            - run: npm ci
            - run: npm test
            - run: npm run test:coverage
            - uses: codecov/codecov-action@v3
              with:
                  files: ./coverage/coverage-final.json

测试 Hook

{
  "hooks": [
    {
      "id": "run-related-tests",
      "event": "PostToolUse",
      "conditions": [
        { "type": "tool", "value": "write_file" },
        { "type": "file", "value": "src/**/*.ts" }
      ],
      "type": "function",
      "function": "async (ctx) => { const file = ctx.data.params.path; const testFile = file.replace('/src/', '/tests/').replace('.ts', '.test.ts'); if (await exists(testFile)) { await runCommand(`npm test -- ${testFile}`); } }"
    }
  ]
}
