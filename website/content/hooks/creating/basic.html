创建基础钩子

来源说明：本文内容基于 Claude Code 官方文档和社区最佳实践整理。

可靠性评级：A 官方文档 + 社区验证

快速开始

创建第一个钩子

# 创建钩子目录
mkdir -p ~/.claude/hooks

# 创建钩子文件
cat > ~/.claude/hooks/my-first-hook.js << 'EOF'
module.exports = async (context) => {
  console.log('钩子被触发了！');
  console.log('事件:', context.event);
  console.log('数据:', context.data);
  
  return { action: 'continue' };
};
EOF

注册钩子

// ~/.claude/hooks/hooks.json
{
  "hooks": {
    "before-command": {
      "handlers": [
        {
          "name": "my-first-hook",
          "script": "my-first-hook.js"
        }
      ]
    }
  }
}

钩子结构

基本结构

// hooks/my-hook.js

/**
 * 钩子处理器
 * @param {Object} context - 钩子上下文
 * @returns {Object} - 返回结果
 */
module.exports = async (context) => {
  // 1. 获取上下文数据
  const { event, data, config } = context;
  
  // 2. 执行钩子逻辑
  // ...
  
  // 3. 返回结果
  return {
    action: 'continue',  // 或 'abort', 'modify'
    data: {}             // 可选：修改后的数据
  };
};

上下文对象

context = {
  event: 'before-command',    // 事件名称
  data: {                     // 事件数据
    command: 'read',
    args: ['file.js']
  },
  config: {},                 // 配置
  session: {},                // 会话信息
  project: {},                // 项目信息
  utils: {}                   // 工具函数
};

返回值

继续执行

// 允许操作继续
return { action: 'continue' };

中断操作

// 阻止操作
return {
  action: 'abort',
  reason: '操作被拒绝'
};

修改数据

// 修改操作数据
return {
  action: 'modify',
  data: {
    ...context.data,
    newArg: 'value'
  }
};

返回错误

// 返回错误
return {
  action: 'error',
  error: new Error('发生错误')
};

基础示例

日志钩子

// hooks/logger.js
const fs = require('fs');
const path = require('path');

module.exports = async (context) => {
  const logFile = path.join(process.env.HOME, '.claude', 'hook.log');
  const logEntry = {
    timestamp: new Date().toISOString(),
    event: context.event,
    data: context.data
  };
  
  fs.appendFileSync(logFile, JSON.stringify(logEntry) + '\n');
  
  return { action: 'continue' };
};

验证钩子

// hooks/validator.js
module.exports = async (context) => {
  const { command, args } = context.data;
  
  // 验证命令参数
  if (command === 'write' && !args[0]) {
    return {
      action: 'abort',
      reason: '写入操作需要指定文件路径'
    };
  }
  
  // 验证文件类型
  const allowedExtensions = ['.js', '.ts', '.jsx', '.tsx'];
  const filePath = args[0];
  const ext = path.extname(filePath);
  
  if (!allowedExtensions.includes(ext)) {
    return {
      action: 'abort',
      reason: `不支持的文件类型: ${ext}`
    };
  }
  
  return { action: 'continue' };
};

通知钩子

// hooks/notifier.js
module.exports = async (context) => {
  const { event, data } = context;
  
  // 发送桌面通知
  if (event === 'on-ready') {
    sendNotification('Claude Code', '已准备就绪');
  }
  
  // 发送邮件通知
  if (event === 'on-error') {
    sendEmail({
      to: 'admin@example.com',
      subject: 'Claude Code 错误',
      body: JSON.stringify(data.error)
    });
  }
  
  return { action: 'continue' };
};

钩子配置

完整配置

{
  "hooks": {
    "before-command": {
      "enabled": true,
      "priority": 100,
      "timeout": 30000,
      "handlers": [
        {
          "name": "validator",
          "script": "hooks/validator.js",
          "enabled": true
        },
        {
          "name": "logger",
          "script": "hooks/logger.js",
          "enabled": true
        }
      ]
    }
  }
}

条件执行

{
  "handlers": [
    {
      "name": "conditional-hook",
      "script": "hooks/conditional.js",
      "condition": {
        "command": ["read", "write"],
        "filePattern": "*.js"
      }
    }
  ]
}

# 触发钩子测试

# 查看钩子日志
> /hook logs my-first-hook
