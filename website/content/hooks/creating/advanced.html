// hooks/chain-hook.js
module.exports = async (context) => {
  const { chain } = context.utils;
  
  // 定义处理链
  const processors = [
    validateInput,
    transformData,
    checkPermissions,
    executeOperation
  ];
  
  // 链式执行
  let result = context.data;
  for (const processor of processors) {
    result = await processor(result);
    if (result.action === 'abort') {
      return result;
    }
  }
  
  return { action: 'continue', data: result };
};

中间件模式

// hooks/middleware-hook.js
const middlewares = [];

// 添加中间件
function use(middleware) {
  middlewares.push(middleware);
}

// 执行中间件
async function execute(context) {
  let index = 0;
  
  async function next() {
    if (index >= middlewares.length) {
      return { action: 'continue' };
    }
    
    const middleware = middlewares[index++];
    return await middleware(context, next);
  }
  
  return await next();
}

// 注册中间件
use(async (ctx, next) => {
  console.log('中间件 1: 前');
  const result = await next();
  console.log('中间件 1: 后');
  return result;
});

use(async (ctx, next) => {
  console.log('中间件 2');
  return await next();
});

module.exports = execute;

规则引擎

// hooks/rule-engine.js
const rules = [
  {
    name: 'block-sensitive-files',
    condition: (ctx) => ctx.data.file.includes('.env'),
    action: 'abort',
    message: '敏感文件禁止访问'
  },
  {
    name: 'require-auth',
    condition: (ctx) => ctx.data.command === 'deploy',
    action: 'checkAuth'
  }
];

module.exports = async (context) => {
  for (const rule of rules) {
    if (rule.condition(context)) {
      if (rule.action === 'abort') {
        return { action: 'abort', reason: rule.message };
      }
      if (rule.action === 'checkAuth') {
        const auth = await checkAuthentication(context);
        if (!auth.valid) {
          return { action: 'abort', reason: '需要认证' };
        }
      }
    }
  }
  
  return { action: 'continue' };
};

状态管理

钩子状态

// hooks/stateful-hook.js
const state = {
  count: 0,
  lastRun: null,
  cache: new Map()
};

module.exports = async (context) => {
  // 更新状态
  state.count++;
  state.lastRun = new Date();
  
  // 使用缓存
  const cacheKey = JSON.stringify(context.data);
  if (state.cache.has(cacheKey)) {
    return state.cache.get(cacheKey);
  }
  
  // 执行操作
  const result = await processContext(context);
  
  // 缓存结果
  state.cache.set(cacheKey, result);
  
  return result;
};

持久化状态

// hooks/persistent-hook.js
const fs = require('fs');
const path = require('path');

const stateFile = path.join(process.env.HOME, '.claude', 'hook-state.json');

async function loadState() {
  try {
    const data = await fs.promises.readFile(stateFile, 'utf8');
    return JSON.parse(data);
  } catch {
    return {};
  }
}

async function saveState(state) {
  await fs.promises.writeFile(stateFile, JSON.stringify(state));
}

module.exports = async (context) => {
  const state = await loadState();
  
  // 使用和更新状态
  state.lastRun = new Date().toISOString();
  state.runCount = (state.runCount || 0) + 1;
  
  await saveState(state);
  
  return { action: 'continue' };
};

错误处理

错误捕获

module.exports = async (context) => {
  try {
    const result = await riskyOperation(context);
    return { action: 'continue', data: result };
  } catch (error) {
    // 记录错误
    console.error('钩子错误:', error);
    
    // 根据错误类型决定处理方式
    if (error.code === 'EACCES') {
      return {
        action: 'abort',
        reason: '权限不足'
      };
    }
    
    if (error.code === 'ENOTFOUND') {
      return {
        action: 'abort',
        reason: '网络连接失败'
      };
    }
    
    // 其他错误
    return {
      action: 'error',
      error
    };
  }
};

重试机制

async function withRetry(fn, options = {}) {
  const { maxRetries = 3, delay = 1000 } = options;
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(r => setTimeout(r, delay * (i + 1)));
    }
  }
}

module.exports = async (context) => {
  const result = await withRetry(
    () => unreliableOperation(context),
    { maxRetries: 5, delay: 2000 }
  );
  
  return { action: 'continue', data: result };
};
