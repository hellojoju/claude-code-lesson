提交前钩子案例

来源说明：本文展示一个完整的提交前钩子实现。

可靠性评级：B 社区实践

钩子概述

提交前钩子在代码提交前自动执行检查，确保代码质量。

功能特性

  • 运行代码检查（Lint）
  • 运行单元测试
  • 检查代码格式
  • 验证提交信息格式

实现代码

// hooks/pre-commit.js
const { execSync } = require('child_process');
const fs = require('fs');
const path = require('path');

module.exports = async (context) => {
  const { data } = context;
  const { files, message } = data;
  
  console.log('🔍 执行提交前检查...\n');
  
  // 1. 检查提交信息格式
  const messageCheck = checkCommitMessage(message);
  if (!messageCheck.valid) {
    return {
      action: 'abort',
      reason: `提交信息格式错误: ${messageCheck.error}`
    };
  }
  
  // 2. 获取暂存的文件
  const stagedFiles = getStagedFiles();
  
  // 3. 运行 Lint 检查
  const lintResult = await runLint(stagedFiles);
  if (!lintResult.success) {
    return {
      action: 'abort',
      reason: `Lint 检查失败:\n${lintResult.errors}`
    };
  }
  
  // 4. 运行测试
  const testResult = await runTests();
  if (!testResult.success) {
    return {
      action: 'abort',
      reason: `测试失败:\n${testResult.errors}`
    };
  }
  
  // 5. 检查敏感信息
  const sensitiveCheck = checkSensitiveInfo(stagedFiles);
  if (!sensitiveCheck.safe) {
    return {
      action: 'abort',
      reason: `发现敏感信息:\n${sensitiveCheck.warnings}`
    };
  }
  
  console.log('✅ 所有检查通过！\n');
  return { action: 'continue' };
};

// 检查提交信息格式
function checkCommitMessage(message) {
  const pattern = /^(feat|fix|docs|style|refactor|test|chore)(\(.+\))?: .{1,50}/;
  
  if (!pattern.test(message)) {
    return {
      valid: false,
      error: '提交信息必须符合 Conventional Commits 格式，例如: feat: 添加新功能'
    };
  }
  
  return { valid: true };
}

// 获取暂存的文件
function getStagedFiles() {
  try {
    const output = execSync('git diff --cached --name-only', { encoding: 'utf8' });
    return output.trim().split('\n').filter(Boolean);
  } catch {
    return [];
  }
}

// 运行 Lint
async function runLint(files) {
  const jsFiles = files.filter(f => /\.(js|jsx|ts|tsx)$/.test(f));
  
  if (jsFiles.length === 0) {
    return { success: true };
  }
  
  try {
    execSync(`npx eslint ${jsFiles.join(' ')}`, { encoding: 'utf8' });
    return { success: true };
  } catch (error) {
    return {
      success: false,
      errors: error.stdout || error.message
    };
  }
}

// 运行测试
async function runTests() {
  try {
    execSync('npm test -- --passWithNoTests', { encoding: 'utf8' });
    return { success: true };
  } catch (error) {
    return {
      success: false,
      errors: error.stdout || error.message
    };
  }
}

// 检查敏感信息
function checkSensitiveInfo(files) {
  const warnings = [];
  const sensitivePatterns = [
    /password\s*=\s*['"][^'"]+['"]/gi,
    /api[_-]?key\s*=\s*['"][^'"]+['"]/gi,
    /secret\s*=\s*['"][^'"]+['"]/gi,
    /token\s*=\s*['"][^'"]+['"]/gi
  ];
  
  for (const file of files) {
    if (!fs.existsSync(file)) continue;
    
    const content = fs.readFileSync(file, 'utf8');
    
    for (const pattern of sensitivePatterns) {
      const matches = content.match(pattern);
      if (matches) {
        warnings.push(`${file}: 发现可能的敏感信息`);
      }
    }
  }
  
  return {
    safe: warnings.length === 0,
    warnings: warnings.join('\n')
  };
}

配置

// hooks/hooks.json
{
  "hooks": {
    "pre-commit": {
      "enabled": true,
      "priority": 100,
      "timeout": 120000,
      "handlers": [
        {
          "name": "pre-commit-checks",
          "script": "pre-commit.js"
        }
      ],
      "config": {
        "skipTests": false,
        "skipLint": false,
        "checkSensitiveInfo": true
      }
    }
  }
}
