自动测试钩子案例

来源说明：本文展示一个完整的自动测试钩子实现。

可靠性评级：B 社区实践

钩子概述

自动测试钩子在文件保存后自动运行相关测试，提供即时反馈。

功能特性

  • 检测文件变更
  • 智能识别相关测试
  • 并行运行测试
  • 实时报告结果

实现代码

// hooks/auto-test.js
const { execSync, spawn } = require('child_process');
const path = require('path');
const chokidar = require('chokidar');

class AutoTestHook {
  constructor() {
    this.testQueue = new Set();
    this.isRunning = false;
    this.watcher = null;
  }
  
  async init(context) {
    // 启动文件监控
    this.watcher = chokidar.watch('./src/**/*.{js,jsx,ts,tsx}', {
      ignored: /node_modules/,
      persistent: true
    });
    
    this.watcher.on('change', (filePath) => {
      this.onFileChange(filePath);
    });
    
    console.log('🔍 自动测试监控已启动');
  }
  
  async onFileChange(filePath) {
    // 查找相关测试文件
    const testFiles = this.findRelatedTests(filePath);
    
    if (testFiles.length === 0) {
      return;
    }
    
    // 添加到测试队列
    testFiles.forEach(f => this.testQueue.add(f));
    
    // 执行测试
    await this.runTests();
  }
  
  findRelatedTests(sourceFile) {
    const tests = [];
    const dir = path.dirname(sourceFile);
    const base = path.basename(sourceFile, path.extname(sourceFile));
    
    // 可能的测试文件位置
    const possibleTests = [
      path.join(dir, `${base}.test.js`),
      path.join(dir, `${base}.test.ts`),
      path.join(dir, `${base}.spec.js`),
      path.join(dir, `${base}.spec.ts`),
      path.join(dir, '__tests__', `${base}.test.js`),
      path.join('tests', sourceFile.replace('src/', '').replace(/\.(js|ts)$/, '.test.js'))
    ];
    
    for (const testFile of possibleTests) {
      if (fs.existsSync(testFile)) {
        tests.push(testFile);
      }
    }
    
    return tests;
  }
  
  async runTests() {
    if (this.isRunning || this.testQueue.size === 0) {
      return;
    }
    
    this.isRunning = true;
    const tests = Array.from(this.testQueue);
    this.testQueue.clear();
    
    console.log(`\n🧪 运行 ${tests.length} 个测试文件...\n`);
    
    try {
      const result = execSync(
        `npx jest ${tests.join(' ')} --passWithNoTests --coverage=false`,
        { encoding: 'utf8' }
      );
      
      console.log('✅ 测试通过\n');
      this.notifyResult(true, result);
    } catch (error) {
      console.log('❌ 测试失败\n');
      console.log(error.stdout);
      this.notifyResult(false, error.stdout);
    } finally {
      this.isRunning = false;
      
      // 检查是否还有待运行的测试
      if (this.testQueue.size > 0) {
        await this.runTests();
      }
    }
  }
  
  notifyResult(success, output) {
    // 发送桌面通知
    const title = success ? '✅ 测试通过' : '❌ 测试失败';
    const notification = {
      title,
      body: success ? '所有测试用例通过' : '部分测试用例失败'
    };
    
    // 可以集成系统通知
    if (process.platform === 'darwin') {
      execSync(`osascript -e 'display notification "${notification.body}" with title "${notification.title}"'`);
    }
  }
  
  async cleanup() {
    if (this.watcher) {
      await this.watcher.close();
    }
  }
}

const hook = new AutoTestHook();

module.exports = async (context) => {
  const { event } = context;
  
  switch (event) {
    case 'on-ready':
      await hook.init(context);
      break;
      
    case 'after-save':
      await hook.onFileChange(context.data.file);
      break;
      
    case 'on-shutdown':
      await hook.cleanup();
      break;
  }
  
  return { action: 'continue' };
};

配置

{
  "hooks": {
    "auto-test": {
      "enabled": true,
      "handlers": [
        {
          "name": "auto-test",
          "script": "auto-test.js",
          "events": ["on-ready", "after-save", "on-shutdown"]
        }
      ],
      "config": {
        "watchPatterns": ["src/**/*.{js,jsx,ts,tsx}"],
        "testPatterns": ["**/*.test.{js,ts}", "**/*.spec.{js,ts}"],
        "runOnSave": true,
        "parallel": true,
        "notifications": true
      }
    }
  }
}
